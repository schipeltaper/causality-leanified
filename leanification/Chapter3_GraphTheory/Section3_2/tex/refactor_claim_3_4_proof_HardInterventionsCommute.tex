\begin{Lem}[Hard interventions commute]\label{hard-interventions-commute}
    Let $G = (J, V, E, L)$ be a CDMG (def \ref{def-cdmg}); throughout we use the notation of def \ref{not-cdmg} and recall the typing and axioms of def \ref{def-cdmg}, namely $J \cap V = \emptyset$, $E \subseteq (J \cup V) \times V$, $L \subseteq V \times V$, $L$ irreflexive ($(v_1, v_2) \in L \Rightarrow v_1 \neq v_2$), and $L$ symmetric ($(v_1, v_2) \in L \Leftrightarrow (v_2, v_1) \in L$). Let
    \[ W_1 \subseteq J \cup V \quad\text{and}\quad W_2 \subseteq J \cup V \]
    be two subsets of nodes (no disjointness between $W_1$ and $W_2$ is assumed, and overlap with $J$ is permitted, as per def \ref{def:G_hard_intervention}'s precondition). All three quantifiers ``for every CDMG $G$'', ``for every $W_1 \subseteq J \cup V$'', and ``for every $W_2 \subseteq J \cup V$'' are read \emph{universally}.

    \emph{Carrier matching for the iterated hard interventions.} The inner two terms of the equality below are iterated hard interventions of the form $(G_{\doit(W_i)})_{\doit(W_j)}$, in which $\doit(W_j)$ is applied to the CDMG $G_{\doit(W_i)}$ produced by def \ref{def:G_hard_intervention}. For this outer application to be well-typed per def \ref{def:G_hard_intervention}'s precondition ``$W \subseteq J \cup V$ of the input CDMG'', we need $W_j$ to be a subset of the carrier of $G_{\doit(W_i)}$, i.e.\
    \[ W_j \;\subseteq\; J_{\doit(W_i)} \cup V_{\doit(W_i)}. \]
    This holds because hard intervention preserves the carrier $J \cup V$ as a set: by items i.\ and ii.\ of def \ref{def:G_hard_intervention},
    \[
        J_{\doit(W_i)} \cup V_{\doit(W_i)} \;=\; (J \cup W_i) \cup (V \sm W_i) \;=\; J \cup V,
    \]
    where the second equality uses $W_i \subseteq J \cup V$ (as in the carrier-matching paragraph of \ref{claim_3_3}). Hence $W_j \subseteq J \cup V = J_{\doit(W_i)} \cup V_{\doit(W_i)}$, so $(G_{\doit(W_i)})_{\doit(W_j)}$ is defined. The same observation applied to $W_1 \cup W_2 \subseteq J \cup V$ shows that the right-hand side $G_{\doit(W_1 \cup W_2)}$ is also defined per def \ref{def:G_hard_intervention}.

    \emph{The conclusion.} The CDMGs $(G_{\doit(W_1)})_{\doit(W_2)}$, $(G_{\doit(W_2)})_{\doit(W_1)}$, and $G_{\doit(W_1 \cup W_2)}$ (all constructed via def \ref{def:G_hard_intervention} as just verified) coincide. Concretely, the LN's displayed triple equality
    \[ \lp G_{\doit(W_1)} \rp_{\doit(W_2)} = \lp G_{\doit(W_2)} \rp_{\doit(W_1)} = G_{\doit(W_1 \cup W_2)} \]
    is the conjunction of the two following equalities of CDMGs:
    \begin{enumerate}[label=(\alph*)]
        \item \emph{Composition with $W_1$ then $W_2$.}
              \[ \lp G_{\doit(W_1)} \rp_{\doit(W_2)} \;=\; G_{\doit(W_1 \cup W_2)}. \]
        \item \emph{Composition with $W_2$ then $W_1$.}
              \[ \lp G_{\doit(W_2)} \rp_{\doit(W_1)} \;=\; G_{\doit(W_1 \cup W_2)}. \]
    \end{enumerate}
    Together (a) and (b) yield the LN's ``symmetry under swap'' reading $\lp G_{\doit(W_1)} \rp_{\doit(W_2)} = \lp G_{\doit(W_2)} \rp_{\doit(W_1)}$ by transitivity of equality. The equality of CDMGs in (a) and (b) is understood componentwise on the four components $(J, V, E, L)$ of def \ref{def-cdmg}, but the conjunctive unpacking into the four field-by-field equalities is deferred to the proof.

      %This shows that our notion:

      %is unambiguous.
\end{Lem}

% Proof body adapted from the LN's \Claude{...} block at
% lecture-notes/lecture_notes/graphs.tex lines 326-355, with three light edits:
% (i) the opening is re-framed to reference the rewrite's (a) and (b) labels,
% (ii) the closing "by symmetry" remark for (b) is expanded into an explicit
% W_1 <-> W_2 swap + commutativity-of-union step, and
% (iii) the bidirected-edge paragraph carries an inline note that, under the
% cdmg_typed_edges refactor's `Sym2 Node` encoding of L, the previously-
% registered deviation `hard_intervention_l_symmetrized_removal` is
% structurally resolved at the def_3_10 row (no symmetrisation step in either
% encoding is needed at this proof's L-component step).
\begin{proof}
We prove sub-claim (a), $\lp G_{\doit(W_1)} \rp_{\doit(W_2)} = G_{\doit(W_1 \cup W_2)}$, by checking the equality of the two CDMGs componentwise on each of the four components $(J, V, E, L)$ of def \ref{def-cdmg} in turn; sub-claim (b), $\lp G_{\doit(W_2)} \rp_{\doit(W_1)} = G_{\doit(W_1 \cup W_2)}$, is then obtained by an explicit symmetry argument at the end of the proof. Both sub-claims rely on the carrier-matching observation already established in the statement block above to license the outer $\doit(\cdot)$ application.

\medskip\noindent\textbf{Proof of (a): composition with $W_1$ then $W_2$.}
We verify the equality $\lp G_{\doit(W_1)} \rp_{\doit(W_2)} = G_{\doit(W_1 \cup W_2)}$ on each of the four components.

\medskip\noindent\emph{Input nodes.}
By item i.\ of def \ref{def:G_hard_intervention} applied twice (first to $G$ with $W_1$, then to $G_{\doit(W_1)}$ with $W_2$),
\[
    J_{\lp G_{\doit(W_1)} \rp_{\doit(W_2)}} \;=\; J_{\doit(W_1)} \cup W_2 \;=\; (J \cup W_1) \cup W_2 \;=\; J \cup (W_1 \cup W_2) \;=\; J_{\doit(W_1 \cup W_2)},
\]
where the middle equality is associativity of $\cup$. \checkmark

\medskip\noindent\emph{Output nodes.}
By item ii.\ of def \ref{def:G_hard_intervention} applied twice,
\[
    V_{\lp G_{\doit(W_1)} \rp_{\doit(W_2)}} \;=\; V_{\doit(W_1)} \sm W_2 \;=\; (V \sm W_1) \sm W_2 \;=\; V \sm (W_1 \cup W_2) \;=\; V_{\doit(W_1 \cup W_2)},
\]
where the middle equality is the standard identity $(A \sm B) \sm C = A \sm (B \cup C)$ for set difference (an element $x$ survives both removals iff $x \notin B$ and $x \notin C$, iff $x \notin B \cup C$). \checkmark

\medskip\noindent\emph{Directed edges.}
By item iii.\ of def \ref{def:G_hard_intervention} applied twice,
\begin{align*}
    E_{\lp G_{\doit(W_1)} \rp_{\doit(W_2)}}
    &= E_{\doit(W_1)} \sm \lC v \tuh w \mid w \in W_2 \rC \\
    &= \lp E \sm \lC v \tuh w \mid w \in W_1 \rC \rp
       \sm \lC v \tuh w \mid w \in W_2 \rC \\
    &= E \sm \lC v \tuh w \mid w \in W_1 \cup W_2 \rC
     = E_{\doit(W_1 \cup W_2)}.
\end{align*}
The penultimate equality is the same set-difference identity used for the output nodes, instantiated at the directed-edge set: an ordered pair $(v, w) \in E$ survives both removal steps iff its head $w$ lies neither in $W_1$ nor in $W_2$, iff $w \notin W_1 \cup W_2$; that condition is exactly the one matching the removal set of a single application of item iii.\ at the union $W_1 \cup W_2$. \checkmark

\medskip\noindent\emph{Bidirected edges.}
By item iv.\ of def \ref{def:G_hard_intervention} applied twice,
\begin{align*}
    L_{\lp G_{\doit(W_1)} \rp_{\doit(W_2)}}
    &= L_{\doit(W_1)} \sm \lC v \huh w \mid w \in W_2 \rC \\
    &= \lp L \sm \lC v \huh w \mid w \in W_1 \rC \rp
       \sm \lC v \huh w \mid w \in W_2 \rC \\
    &= L \sm \lC v \huh w \mid w \in W_1 \cup W_2 \rC
     = L_{\doit(W_1 \cup W_2)}.
\end{align*}
The penultimate equality combines the two removal clauses by the same union-of-conditions step used for the directed-edge case, with $v \huh w$ in place of $v \tuh w$. \checkmark

\smallskip\noindent\emph{Remark on the Lean encoding of $L$ under the \texttt{cdmg\_typed\_edges} refactor.} Under the post-refactor Lean encoding of def \ref{def-cdmg} (the \texttt{cdmg\_typed\_edges} refactor; see \texttt{leanification/refactors/refactor\_cdmg\_typed\_edges.md} for the design block), $L$ is typed as a $\Finset$ of \emph{unordered} pairs $\Sym_2(\Node)$ rather than as a $\Finset$ of ordered pairs $\Node \times \Node$ paired with a separate $\Sym$-symmetry axiom. Consequently, item iv.\ of def \ref{def:G_hard_intervention} reads literally: a bidirected edge $s \in L$ (an unordered pair) is removed iff some endpoint $v$ of $s$ lies in $W$ (kept-condition: $\forall v \in s,\ v \notin W$). There is no ``second component'' on an unordered pair to test in isolation, so the LN's literal one-sided phrasing $v_2 \in W$ collapses to ``any endpoint in $W$'', and no symmetrisation step is required to satisfy any CDMG axiom (swap-symmetry is definitional on $\Sym_2$, so $s(v_1, v_2) = s(v_2, v_1)$ holds by construction). The previously-registered deviation \emph{hard\_intervention\_l\_symmetrized\_removal} -- attached to the pre-refactor encoding of def \ref{def:G_hard_intervention} -- is therefore \emph{structurally resolved at the def\_3\_10 row} under the refactor, and does not surface at this row. The composition step of this proof, expressed at the Lean level, reads as ``filter-by-$p_1$ then filter-by-$p_2 = $ filter-by-$(p_1 \wedge p_2)$'' with $p_i(s) := (\forall v \in s,\ v \notin W_i)$: applied at the union $W_1 \cup W_2$, the pointwise predicate equivalence
\[
    (\forall v \in s,\ v \notin W_1) \wedge (\forall v \in s,\ v \notin W_2) \;\Longleftrightarrow\; \forall v \in s,\ v \notin W_1 \cup W_2
\]
follows by pulling the conjunction inside the bounded universal (each disjoint-$v$ instance is one $\wedge$-pair) plus the membership-in-union rewrite $v \notin W_1 \cup W_2 \Leftrightarrow v \notin W_1 \wedge v \notin W_2$. Both the LN's set-difference algebra in the displayed line and the Lean-level composition step land on the same final bidirected-edge set $L_{\doit(W_1 \cup W_2)}$.

\smallskip
Combining the four componentwise equalities above, the CDMGs $\lp G_{\doit(W_1)} \rp_{\doit(W_2)}$ and $G_{\doit(W_1 \cup W_2)}$ agree field by field on each of $(J, V, E, L)$ and are therefore equal as CDMGs (def \ref{def-cdmg}). This establishes sub-claim (a).

\medskip\noindent\textbf{Proof of (b): composition with $W_2$ then $W_1$.}
Sub-claim (b) is obtained by swapping the roles of $W_1$ and $W_2$ in the argument for (a). Concretely, re-running the four-component check above with the substitution $W_1 \leftrightarrow W_2$ -- the proof depends on the pair of subsets only through their symbolic roles, and the displayed identities $(J \cup W_1) \cup W_2 = J \cup (W_1 \cup W_2)$, $(V \sm W_1) \sm W_2 = V \sm (W_1 \cup W_2)$, and the analogous edge-set identities all retain their truth under the swap -- yields
\[
    \lp G_{\doit(W_2)} \rp_{\doit(W_1)} \;=\; G_{\doit(W_2 \cup W_1)}.
\]
This is precisely sub-claim (a) instantiated at the swapped pair $(W_2, W_1)$. We emphasise that sub-claim (b) is \emph{not} a literal instance of (a): the iterated hard intervention on the left-hand side of (b) is built from the swapped pair $(W_2, W_1)$, and the right-hand-side union appearing in (a) for that swapped pair is $W_2 \cup W_1$, not the $W_1 \cup W_2$ in the statement of (b). The two right-hand sides agree by commutativity of union, $W_2 \cup W_1 = W_1 \cup W_2$, which gives
\[
    G_{\doit(W_2 \cup W_1)} \;=\; G_{\doit(W_1 \cup W_2)}
\]
as CDMGs: def \ref{def:G_hard_intervention} depends on $W$ only through the underlying set $W \subseteq J \cup V$ (each of items i.\ -- iv.\ refers to $W$ only via the operations $J \cup W$, $V \sm W$, and the membership predicate ``$\cdot \in W$''), and equal sets produce equal CDMGs componentwise. Chaining the two equalities,
\[
    \lp G_{\doit(W_2)} \rp_{\doit(W_1)} \;=\; G_{\doit(W_2 \cup W_1)} \;=\; G_{\doit(W_1 \cup W_2)},
\]
establishes sub-claim (b).

\medskip\noindent\emph{Recovery of the LN's triple equality.}
By sub-claims (a) and (b) -- both of which equate their respective iterated hard intervention with the single right-hand side $G_{\doit(W_1 \cup W_2)}$ -- transitivity of equality on that shared right-hand side yields
\[
    \lp G_{\doit(W_1)} \rp_{\doit(W_2)} \;=\; G_{\doit(W_1 \cup W_2)} \;=\; \lp G_{\doit(W_2)} \rp_{\doit(W_1)},
\]
which is the LN's stated triple equality and in particular recovers the swap-symmetry reading $\lp G_{\doit(W_1)} \rp_{\doit(W_2)} = \lp G_{\doit(W_2)} \rp_{\doit(W_1)}$.
\end{proof}
