\begin{Def}[Causal Bayesian network]
    \label{def-CBN}
    A \emph{causal Bayesian network (CBN)}---by definition---consists of:
    \begin{enumerate}[label=\alph*)]
        \item a conditional directed acyclic graph (CDAG): $G = (J,V,E)$ (with finite vertex sets, and no bidirected edges),
        \item a standard measurable space $\Xcal_v$ for every $v \in J \cup V$,
        \item for every $v \in V$, a Markov kernel: 
            $\displaystyle  P_v\lp X_v| X_{\Pa^G(v)}\rp$:
            %$\displaystyle  P_v\lp X_v|\doit\lp X_{\Pa^G(v)}\rp\rp$:
        \[\begin{array}{ccrcl}
           & & \Xcal_{\Pa^G(v)} & \dshto & \Xcal_v, \\
           && \qquad (A,x_{\Pa^G(v)}) & \mapsto &  P_v\lp X_v \in A|X_{\Pa^G(v)} = x_{\Pa^G(v)}\rp,
           %&& \qquad (A,x_{\Pa^G(v)}) & \mapsto &  P_v\lp X_v \in A|\doit\lp X_{\Pa^G(v)} = x_{\Pa^G(v)} \rp\rp,
        \end{array}\]
        where we write for $D \ins J \cup V$:
               \begin{align*} 
                   \Xcal_D &:= \prod_{v \in D} \Xcal_v,&
                   \Xcal_\emptyset &:= \Asterisk = \{\ast\},\\
                   X_D &:= (X_v)_{v \in D},&
                   X_\emptyset &:= \ast,\\
                   x_D &:= (x_v)_{v \in D},&
                   x_\emptyset &:= \ast.
              \end{align*}
    \end{enumerate}
    %By abuse  of notation, we will abbreviate the tupel of Markov kernels as: 
    %\[\Pr = \lp  P_v\lp X_v|\doit\lp X_{\Pa^G(v)}\rp\rp \rp_{v \in V}, \] 
    %keep the spaces $\Xcal_v$ implicit and refer to the tupel $(G,\Pr)$ as the CBN.
\end{Def}
\begin{Rem}
  Most existing accounts of causal Bayesian networks do not formally distinguish input nodes from output nodes.
  The reasons that we do make this disctinction are of a measure-theoretical nature.
  If all variables are discrete, and all probability mass functions and Markov kernels are strictly positive, then the formal differences between input and output nodes may be ignored and everything can be considered as output nodes.
\end{Rem}

\begin{Def}[The joint Markov kernel of a causal Bayesian network with input variables]
    Consider a causal Bayesian network with input variables with CDAG $G=(J,V,E)$ with Markov kernels
    $P_v\lp X_v| X_{\Pa^G(v)}\rp$ for $v \in V$.
    For a fixed topological ordering $<$ of $G$ 
    we  then define the joint Markov kernel of the CBN:
    \[ \Xcal_J \dshto \Xcal_V\]
    as follows:
    \[  P(X_V|\doit(X_J)) := \bigotimes_{v \in V}^> P_v\lp X_v|X_{\Pa^G(v)}\rp,  \]
    where the nodes $v$ run through $V$ in reverse ordering of $<$, i.e.\ 
    all parents are on the right of all their children.
\end{Def}

\begin{Rem}[Marginalization and conditioning]
    Let $P(X_V|\doit(X_J))$ be the joint Markov kernel of a CBN.
    We can extend it to a joint Markov kernel including $X_J$:
    \[P(X_V,X_J|\doit(X_J)) = P(X_V|\doit(X_J)) \otimes \delta(X_J|X_J).  \]
    For any $A, B \ins J \cup V$ we then also have the \emph{marginal conditional} Markov kernel:
    \[P(X_A|X_B,\doit(X_J)),\]
    which exists by theorem \ref{thm-regular-conditional-Markov-kernel} due to the use of standard measurable spaces
    and is unique up to a $P(X_B|\doit(X_J))$-null set.\\
    Furthermore, if $C \ins J$ and we have: 
    \[X_A \Indep_{P(X_V|\doit(X_J))} X_J \given X_C,\]
    then we also have a Markov kernel:
    \[ P(X_A|\doit(X_C)) \]
    that fits into the equation:
    \[ P(X_A,X_C|\doit(X_J)) = P(X_A|\doit(X_C)) \otimes P(X_C|\doit(X_J)). \]
    Note that this $P(X_A|\doit(X_C))$ is unique up to a $P(X_C|\doit(X_J))$-null set.
    Since, further, $P(X_C|\doit(X_J)) = \delta(X_C|X_J)$, 
    we even get that $P(X_A|\doit(X_C))$ is unique (not just up to null sets).
    In other words, the above conditional independence states that $P(X_A|\doit(X_J))$ is only dependent on the arguments from $X_C$
    and can be represented by a Markov kernel $P(X_A|\doit(X_C))$. 
\end{Rem}

\begin{Def}[Causal Bayesian network with latent variables]\label{def:il-cbn}
    A \emph{causal Bayesian network with latent variables (L-CBN)}--- per 
    definition---consists of a CBN:
    \[ M= \lp G^+=(J,V^+,E^+), \lp P_v(X_v|X_{\Pa^{G^+}(v)}) \rp_{v \in V^+} \rp, \]
    together with a disjoint decomposition of the output nodes $V^+=V \dcup U$ into 
    \emph{observed nodes} $V$ and \emph{unobserved nodes} $U$.
\end{Def}


\begin{Rem}
    In the Definition \ref{def:il-cbn} we make the distinction between the set of observed nodes $V$ and that of unobserved nodes $U$.
    We could have made that distinction already earlier in the graph theory chapters and introduce CDAGs
    $G^+ = \lp J, (V,U),E^+ \rp$, where we make the distinction between these node types part of the (or a new) definition.
    However, most of the time these sets are mathematically treated the same way and we could just consider 
    their union $V^+= V \dcup U$. Usually the distinction between $V$ and $U$ is only made to indicate which variables are marginalized out.
    Also, it often happens that one considers the same CBN in different settings, and which variables are observed and unobserved depends on the setting (for example, during training of a classifier both features and labels are observed, while during testing only features are observed).
    For all these reasons, we do not consider the specification of which variables are observed and which are latent part of the model.
\end{Rem}

\subsection{Global Markov Property}

\begin{Thm}[Global Markov property for causal Bayesian networks]
\label{thm-gmp-mCBN}
Consider a causal Bayesian network $M$  with \observational CADMG $G=(J,V,E,L)$ 
and \observational Markov kernel $P(X_V|\doit(X_J))$.
Then for all $A, B, C \ins J \cup V$ (not necessarily disjoint) we have the implication:
\[ A \idPerp_G B \given C \qquad \implies \qquad X_A \Indep_{P(X_V|\doit(X_J))} X_B \given X_C.   \]
\end{Thm}

\begin{Rem}
    If one wants to make the implicit dependence on $J$ in Theorem \ref{thm-gmp-mCBN} 
    more explicit one can equivalently also write:
\[ A \idPerp_G J \cup B \given C \qquad \implies \qquad X_A \Indep_{P(X_V|\doit(X_J))} X_J,X_{B} \given X_C.   \]
\end{Rem}

\begin{Not}
    Let $A,B,C \ins J \cup V$ with $X_A \Indep_{P(X_V|\doit(X_J))} X_B \given X_C$, then we have a factorization:
    \[ P(X_A,X_B,X_C|\doit(X_J)) = Q(X_A|X_C) \otimes P(X_B,X_C|\doit(X_J)),\]
    for some Markov kernel: $Q(X_A|X_C)$. If we marginalize out $X_B$ and the deterministic $X_{C \cap J}$,
    we get:
    \[ P(X_A,X_{C\cap V}|\doit(X_J)) = Q(X_A|X_C) \otimes P(X_{C\cap V}|\doit(X_J)).\]
    So we see that $Q(X_A|X_C)$ is a conditional Markov kernel:
    \[P(X_A|X_{C\cap V},\doit(X_J)) \]
    that does only depend on $X_{J \cap C}$ in the $\doit$-part.
    So we will use the following notation for $Q(X_A|X_C)$ (or in any other order behind the conditioning line):
    \[ P(X_A|X_{C \cap V},\doit(X_{C \cap J}),\cancel{\doit(X_J)}) := Q(X_A|X_C).  \]
    Note that by Theorem~\ref{thm-ci-equivalences} we may (but do not need to) explicitly mention $X_B$ as in:
    \[ P(X_A|\cancel{X_B},X_{C \cap V},\doit(X_{C \cap J}),\cancel{\doit(X_J)}),  \]
    because the Markov kernels are almost surely equal:
  \[ P(X_A|X_{C \cap V},\doit(X_{C \cap J}),\cancel{\doit(X_J)}) = P(X_A|\cancel{X_B},X_{C \cap V},\doit(X_{C \cap J}),\cancel{\doit(X_J)}) \qquad P(X_C|X_J)\text{-a.s.}.\]
    In these suggestive notations we can state the global Markov property (Theorem \ref{thm-gmp-mCBN}) as:
    \begin{align*} 
    &A \idPerp_G B \given C \\
    &\implies \; P(X_A|X_B,X_C,\doit(X_J)) \\
      & \qquad\qquad = P(X_A|\cancel{X_B},X_{C \cap V},\doit(X_{C \cap J}),\cancel{\doit(X_J)})) && P(X_B,X_C|\doit(X_J))\text{-a.s.} \\
      & \qquad\qquad = P(X_A|X_{C \cap V},\doit(X_{C \cap J}),\cancel{\doit(X_J)}) && P(X_B,X_C|\doit(X_J))\text{-a.s.}. 
    \end{align*}
\end{Not}

\subsection{Operations on Causal Bayesian Networks}
\subsubsection{Hard Interventions on Causal Bayesian Networks}

\begin{Def}[Hard intervention on causal Bayesian network]
    Consider a causal Bayesian network (CBN) given by: 
        \[ M= \lp G=(J,V,E), \lp P_v(X_v|X_{\Pa^G(v)}) \rp_{v \in V} \rp. \]
   % $\lp G, P\lp X_V|\doit(X_J)\rp\rp$  with CDAG: $G=(J,V,E)$ and Markov kernels: $P_v\lp X_v|\doit\lp X_{\Pa^G(v)}\rp\rp$ for $v \in V$.
    Now let  $ W \ins J \cup V$ be any subset.
    Then we define the \emph{intervened causal Bayesian network} w.r.t.\ $W$ via:
    \begin{enumerate}
        \item CDAG: $G_{\doit(W)} = (J \cup W, V \sm W, E_{\doit(W)}) $, and:
        \item Markov kernels: $P_v(X_v|X_{\Pa^G(v)})$ for $v \in V \sm W$.
    \end{enumerate}
    Its \observational Markov kernel is then:
    \[ P(X_{V \sm W}|\doit(X_{J \cup W})) = \bigotimes_{v \in V\sm W} P_v(X_v|X_{\Pa^G(v)}).  \]
    Note that if $ v\in V \sm W$ then $\Pa^G(v) = \Pa^{G_{\doit(W)}}(v)$.
\end{Def}

\begin{Rem}
    Note that the above notations imply for every $v \in V$ and $W \ins V \sm \{ v \}$ the identifications:
    \[ P_v(X_v|X_{\Pa^G(v)}) =  P(X_v|\doit(X_{J \cup V \sm W})) = P(X_v|\doit(X_{\Pa^G(v)})), \]
    which we will use interchangably in the following.
\end{Rem}


\begin{Rem}[Hard intervention on causal Bayesian network with latent variables]
We define hard interventions on an L-CBN the same way as on a CBN, but we usually 
only allow for interventions on sets $W \ins J \cup V$, i.e.\ with $W \cap U = \emptyset$, where $U$ is the set of latent variables.
\end{Rem}

\subsubsection{Node-Splitting on Causal Bayesian Networks}

\begin{Def}[Node-splitting on causal Bayesian network]
    Consider a causal Bayesian network (CBN) given by $\lp G, P\lp X_V|\doit(X_J)\rp\rp$ 
    with CDAG: $G=(J,V,E)$ and Markov kernels: $P_v(X_v|X_{\Pa^G(v)})$ for $v \in V$.
    Now let  $ W \ins V$ be any subset.
    Then we define the \emph{node-split CBN w.r.t.\ $W$} as the causal Bayesian network given by:
    \begin{enumerate}
      \item CDAG: $G':=G_{\spl(W)} = (J, (V\sm W) \cup W^0 \cup W^1,E_{\spl(W)})$, and:
        \item Markov kernels for $v \in V$:
            \[ P_{v^0}(X_{v^o} \in A| X_{\Pa^{G'}(v^o)}=\tilde x) := P_v(X_v \in A|X_{\Pa^G(v)}=\tilde x), \]
          where for brevity we put $v^o:=v$ for $v \in V \sm W$, and for $w^1 \in W^1$:
            \[ P_{w^1}(X_{w^1} \in A| X_{w^0}=\tilde x) := \delta(X_{w^1} \in A|X_{w^0}=\tilde x). \]
    \end{enumerate}
\end{Def}

\begin{Rem}
    Similarly, we can define node-splitting interventions on causal Bayesian network with latent variables, 
    but allow only $W$ with $W \cap U = \emptyset$.
\end{Rem}

This operation can be used to reason about ``single-world'' counterfactuals using CBNs.
We will not dwell on the details here, but discuss this in Chapter~\ref{sec:counterfactuals} in the context of SCMs.

\subsubsection{Node-Splitting Hard Interventions on Causal Bayesian Networks}

\begin{Def}[Node-splitting hard intervention on causal Bayesian network]
    Consider a causal Bayesian network (CBN) given by $\lp G, P\lp X_V|\doit(X_J)\rp\rp$ 
    with CDAG: $G=(J,V,E)$ and Markov kernels: $P_v(X_v|X_{\Pa^G(v)})$ for $v \in V$.
    Now let  $ W \ins V$ be any subset.
    Then we define the \emph{node-splitting hard intervention} w.r.t.\ $W$ as the causal Bayesian network given by:
    \begin{enumerate}
        \item CDAG: $G':=G_{\swig(W)} = (J \dcup W^i, W^o \dcup V\sm W,E_{\swig(W)})$, and:
        \item Markov kernels for $v \in V$:
            \[ P_{v^o}(X_{v^o} \in A| X_{\Pa^{G'}(v^o)}=\tilde x) := P_v(X_v \in A|X_{\Pa^G(v)}=\tilde x), \]
            %\[ P_{v^o}\lp X_{v^o} \in A|\doit\lp X_{\Pa^{G'}(v^o)}=\tilde x\rp\rp := P_v\lp X_v \in A|\doit\lp X_{\Pa^G(v)}=\tilde x\rp\rp, \]
         where for brevity we put $v^o:=v$ for $v \in V \sm W$.
    \end{enumerate}
\end{Def}

\begin{Rem}
    Similarly, we can define node-splitting hard interventions on causal Bayesian network with latent variables, 
    but allow only $W$ with $W \cap U = \emptyset$.
\end{Rem}


\subsubsection{Soft Interventions on Causal Bayesian Networks}

\begin{Rem}[Modelling soft interventions on causal Bayesian networks]
    \label{CBN-soft-int}
    Consider a  causal Bayesian network given by $\lp G, P\lp X_V|\doit(X_J)\rp\rp$ 
    with CDAG: $G=(J,V,E)$ and Markov kernels: $P_v(X_v|X_{\Pa^G(v)})$ for $v \in V$.\\
    Let $W \ins J \cup V$. %\Joris{I would suggest $W \ins V$, it doesn't seem to make sense to model a soft intervention on an input node. Patrick wants to keep it, though, as he believes it simplified some statements later on for the do-calculus.}
    In order to model a soft intervention on variables $X_w$ for $w \in W\sm J$, we introduce 
    intervention nodes $I_w \tuh w$ for $w \in W \sm J$, which come with new input variables $X_{I_w}$, 
    and replace 
    the Markov kernel: 
    \[P_w(X_w|X_{\Pa^G(w)})\]
    %\[P_w\lp X_w|\doit\lp X_{\Pa^G(w)}\rp\rp\]
    for $w \in W\sm J$ by one that models the dependence on the 
    soft intervention variables properly:
    \[  P_w(X_w|X_{\Pa^G(w)},X_{I_w}). \]

    \begin{enumerate}
        \item $\Xcal_{I_w} := \Xcal_w \dcup \{\star\}$,

        %\item $ P_w\lp X_w \in A|\doit\lp X_{\Pa^G(w)}=x_{\Pa^G(w)} ,X_{I_w}=x_{I_w}\rp\rp : =$
        \item $ P_w(X_w \in A|X_{\Pa^G(w)}=x_{\Pa^G(w)} ,X_{I_w}=x_{I_w}) : =$

\end{Rem}

\subsubsection{Marginalization of Causal Bayesian Networks}

\begin{Def}[Marginalization of causal Bayesian network with latent variables]
    Consider a causal Bayesian network with latent variables (L-CBN): 
     \[ M= \lp G^+=(J,(V,U),E^+), \lp P_v(X_v|X_{\Pa^{G^+}(v)})\rp_{v \in V \cup U}\rp,  \]
    Let $W \ins V$ be a subset.
    We then define the \emph{marginalized L-CBN} by just replacing $V$ with $V \sm W$ and $U$ with $U \dcup W$. 
    The Markov kernels $P_v$ for $v \in V \dcup U = (V \sm W) \dcup (U \dcup W)$ stay the same.\\
    With this definition the \observational Markov kernel marginalizes to:
    \[ P\lp X_{V\sm W}|\doit\lp X_J\rp\rp,  \]
    and the \observational CADMG becomes:
    \[ (G^+)^{\sm (U \dcup W)} = G^{\sm W},\]
    i.e.\ the marginalized $G$ w.r.t.\ $W$.
\end{Def}

\subsection{Standard Forms of Causal Bayesian Networks}

\begin{Def}
    Consider two  causal Bayesian network with latent variables (L-CBNs):
    \begin{align*}
        M_1 &= \lp G^+_1=\lp J_1,(V_1,U_1),E^+_1 \rp, \lp P_{1,v}( X_v|X_{\Pa^{G^+_1}(v)}) \rp_{v \in U_1 \cup V_1} \rp,\\
        M_2 &= \lp G^+_2=\lp J_2,(V_2,U_2),E^+_2 \rp, \lp P_{2,v} ( X_v|X_{\Pa^{G^+_2}(v)}) \rp_{v \in U_2 \cup V_2} \rp.
    \end{align*}

    We call them \emph{interventionally equivalent} if all of the following conditions hold:
    \begin{enumerate}
        \item $J_1 = J_2 =:J$,
        \item $V_1 = V_2 =:V$,
        \item $\Xcal_{1,v} = \Xcal_{2,v} =: \Xcal_v$ for all $v \in J \cup V$,
        \item for all subsets $W \ins V$ we have the equality of the intervened Markov kernels:
            \[P_1\lp X_{V \sm W}| \doit\lp X_{J \cup W} \rp\rp  = P_2\lp X_{V\sm W}| \doit\lp X_{J \cup W} \rp\rp.\]
    \end{enumerate}
\end{Def}

\begin{Def}[Cliques and maximal cliques of undirected graphs]
    Let $G=(V,L)$ be an undirected graph.
    A set of nodes $W \ins V$ is called a \emph{clique}\footnote{A clique is also called \emph{complete subgraph} in the literature.} of $G$ if for all $w_1,w_2 \in W$ with $w_1 \neq w_2$ we have that the edge $w_1 \tut w_2 \in L$.
    A clique $W$ is called a \emph{maximal clique} of $G$ if for every clique $\tilde W$ of $G$ with $W \ins \tilde W$ we have that
    $W=\tilde W$.
\end{Def}


\begin{DefThm}[Standard forms of L-CBNs]
    \label{bcn-scm}
    \label{bcn-standard-form}
Consider a causal Bayesian network $M$ with  latent variables (L-CBN) 
with \observational CADMG $G=(J,V,E,L)$. Let
\begin{align*}
    \Ccal &:=\lC W \ins V \st W \text{ maximal clique of } (V,L) \rC.
\end{align*}
the sets of all maximal cliques 
of the (undirected) graph consisting only of the nodes from $V$ and the bidirected edges from $G$.
Define the set of (latent) nodes:
\[ \tilde U := \lC \tilde u_W \st W \in \Ccal \rC,\]
and directed edges:
\[ \tilde E^+ := E \dcup \lC \tilde u_W \tuh w\st W \in \Ccal, w \in W\rC.\]
Then there exists an L-CBN of the form:
\[\tilde M= \lp \tilde G^+=\lp J,(V,\tilde U),\tilde E^+ \rp, \lp \tilde P_v ( X_v|X_{\Pa^{\tilde G^+}(v)}) \rp_{v \in V \cup \tilde U} \rp\]
that is interventionally equivalent to $M$.
Furthermore, we can choose to arrange them in \emph{one} of the following ways:
\begin{enumerate}
    \item \emph{Structural causal model form}: All Markov kernels for $v \in V$ are deterministic:
\[ \tilde P_v(X_v \in A|X_{\Pa^{G^+}(v)}=\tilde x) =  \delta(R_v \in A|X_{\Pa^{G^+}(v)}=\tilde x), \]
%\[ \tilde P_v(X_v \in A|\doit(X_{\Pa^{G^+}(v)}=\tilde x)) =  \delta(R_v \in A|X_{\Pa^{G^+}(v)}=\tilde x), \]
for some measurable maps $R_v$, $v \in V$. OR:
    \item \emph{Canonical form}: All latent variables $\tilde u_W$ with $\#W = 1$ and the corresponding variables, edges and Markov kernels can be removed from $\tilde M$ as well, leaving us only with the latent variables $\tilde u_W$ with $W \in \Ccal$ and $\#W \ge 2$. 
\end{enumerate}
%We will call such L-CBNs $\tilde M$ a \emph{structural causal model form} (case 1) and \emph{standard form} (case 2), resp., of the original L-CBN $M$.
\end{DefThm}

\begin{Rem} Consider the standard forms $\tilde M$ of $M$ from Definition/Theorem \ref{bcn-standard-form}.
    \begin{enumerate}
        \item In particular, we have:
    \begin{enumerate}
      \item $(\tilde G^+)^{\sm \tilde U} = G$,
     \item $\Pa^{\tilde G^+}(\tilde U) = \emptyset$,
     \item $\Ch^{\tilde G^+}(u) \in \Ccal$ for every $u \in \tilde U$.
    \end{enumerate}
  \item We can use measurable embeddings/isomorphisms: $\Xcal_u \inj [0,1]$ for $u \in \tilde U$ to further restrict to the case:
\begin{enumerate}
    \item $\Xcal_u \cong [0,1]$,
    \item $\tilde P_u(X_u)$ is the uniform distribution on $[0,1]$.
\end{enumerate}
    \item Note that the Markov kernels dependent on $X_{\tilde U}$ might not be unique as we can always transform $[0,1]$ to $[0,1]$ in strange ways.
    \item The construction of the canonical form generally\footnote{Excluding degenerate L-CBNs. In those cases one could possibly remove even more latent variables. \label{ref:can-form-deg}} leads to an interventionally equivalent L-CBN with the smallest number of latent variables such that its \observational CADMG stays unchanged. 
    \item The construction of the structural causal model form generally\footref{ref:can-form-deg} leads to an interventionally equivalent L-CBN with the smallest number of latent variables such that its \observational CADMG stays unchanged and such that every Markov kernel with non-trivial input is deterministic. 
    \end{enumerate}
\end{Rem}

\begin{Rem}[Marginalizations and hard interventions on standard forms]
    Let the following L-CBN be in one of the standard forms:    
    \[ \lp G^+=(J,V,U,E^+), \; P\lp X_{V \cup U}| \doit\lp X_{J} \rp\rp \rp.\]
    Now let $W \ins V$ then we defined the marginalization w.r.t.\ $W$ by replacing $V$ with $V \sm W$ and $U$ with $U \cup W$.
    We could re-define the marginalization as a corresponding standard form of that procedure.\\
    Similarly we could post-process hard interventions with standardization steps.
\end{Rem}

    %This is mostly due to the fact that disjoint marginalization and interventions commute.
\end{proof}
\end{Thm}



\subsubsection{Standard Form of Causal Bayesian Networks}





\begin{DefThm}[Standard forms of L-CBNs]
    \label{bcn-standard-form}
Consider a causal Bayesian network with input variables and with latent variables (L-CBN):
     \[ M = \lp G^+=(J,V,U,E^+), \; P\lp X_{V}, X_{U}| \doit\lp X_{J} \rp\rp \rp.\]
Then there exists an interventionally equivalent L-CBN:
\[ \tilde M = \lp \tilde G^+=(J,V,\tilde U,\tilde E^+), \; \tilde P\lp X_{V}, X_{\tilde U}| \doit\lp X_{J} \rp\rp \rp,\]
such that for every $u, u_1, u_2 \in \tilde U$ with $u_1 \neq u_2$:
\begin{enumerate}
    \item $\Pa^{\tilde G^+}(u) = \emptyset$,
    \item $\#\Ch^{\tilde G^+}(u) \ge 2$,
    \item $\Ch^{\tilde G^+}(u_1) \nsubseteq \Ch^{\tilde G^+}(u_2)$,
    \item $G:=(G^+)^{\sm U} = (\tilde G^+)^{\sm \tilde U}$.
\end{enumerate}
Furthermore, the CDAG $\tilde G^+$ is uniquely determined by $G^+$.
%and $\tilde G^+$ and $G^+$ have the same observational CADMG $G$.
We will call such a L-CBN a \emph{standard form} of the original L-CBN.\\
%The other aspects like the Markov kernels dependent on the $X_U$ of the latent space might not be unique.\\
Furthermore, if we are willing to use measurable embeddings/isomorphisms: $\Xcal_u \inj [0,1]$, then we 
can further restrict for $u \in \tilde U$ to:
\begin{enumerate}[resume]
    \item $\Xcal_u \cong [0,1]$
    \item $\tilde P_u(X_u)$ is the uniform distribution on $[0,1]$.
\end{enumerate}
Note that in any case the Markov kernels dependent on $X_U$ might not be unique as we can always transform $[0,1]$ to $[0,1]$ in strange ways.
\begin{proof}
    This follows from the SCM representation of LCCMNs, theorem \ref{bcn-scm}, 
    by marginalizing over all $X_u$ with $\#\Ch^{\tilde G^+}(u) \le 1$, i.e.\ by replacing the left (deterministic) 
    Markov kernel dependent on 
    $X_u$ in the product:
    \[P_v(X_v|\doit(X_{\Pa^{\tilde G^+}(v) \sm \{u\}}, X_u)) \otimes P_u(X_u),\]
    by the composition:
    \[P_v(X_v|\doit(X_{\Pa^{\tilde G^+}(v) \sm \{u\}}, X_u)) \circ P_u(X_u),\]
    which might not be deterministic anymore.
\end{proof}
\end{DefThm}
