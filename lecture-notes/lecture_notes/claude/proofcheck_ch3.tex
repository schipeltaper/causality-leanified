\section{Proof 1: Acyclicity $\iff$ Topological order (lines 198--214)}
%% ========================================================================

\textbf{Statement.} A CDMG $G$ is acyclic iff it has a topological order.

\textbf{Analysis.}
\begin{enumerate}[leftmargin=*]
  \item $(\Leftarrow)$: If a topological order $<$ exists and there were a non-trivial
    directed walk $v_0 \to v_1 \to \cdots \to v_n = v_0$ ($n\ge 1$),
    then $v_0 < v_1 < \cdots < v_n = v_0$, contradiction.  $\checkmark$
  \item $(\Rightarrow)$: Inductive construction by repeatedly selecting a node with
    no parents in the remaining induced subgraph.  Such a node exists because:
    an acyclic finite graph has a node with no parents (otherwise following parent
    edges would produce a directed cycle).  $\checkmark$

    Correctness of the order: if $v_j \in \Pa^G(v_i)$, then when $v_i$ was
    selected, $v_j$ was either already removed ($j < i$) or still present
    (contradicting $\Pa^{G_i}(v_i)=\emptyset$).  $\checkmark$
\end{enumerate}

\textbf{Verdict: CORRECT.}

\section{Proof 2: Hard interventions commute (lines 285--313)}
%% ========================================================================

\textbf{Statement (Lem.\ \texttt{hard-interventions-commute}).}
$(G_{\doit(W_1)})_{\doit(W_2)} = (G_{\doit(W_2)})_{\doit(W_1)} = G_{\doit(W_1\cup W_2)}$.

\textbf{Analysis.}
Verified component by component:
\begin{itemize}
  \item Input nodes: $(J\cup W_1)\cup W_2 = J\cup(W_1\cup W_2)$.  $\checkmark$
  \item Output nodes: $(V\setminus W_1)\setminus W_2 = V\setminus(W_1\cup W_2)$.  $\checkmark$
  \item Directed edges: $(E\setminus\{v\to w:w\in W_1\})\setminus\{v\to w:w\in W_2\}
    = E\setminus\{v\to w:w\in W_1\cup W_2\}$.  $\checkmark$
  \item Bidirected edges: same pattern.  $\checkmark$
\end{itemize}

\textbf{Verdict: CORRECT.}

\section{Proof 3: Bifurcations alternative (lines 321--326)}
%% ========================================================================

\textbf{Statement (Prp.\ \texttt{prp:bifurcations\_alternative}).}
A bifurcation between $v$ and $w$ with source $c$ exists iff $v\ne w$ and
$c\in\Anc^{G_{\doit(w)}}(v)\setminus\{v\}$ and $c\in\Anc^{G_{\doit(v)}}(w)\setminus\{w\}$.

\textbf{Analysis.}
$(\Rightarrow)$: A bifurcation $v\leftarrow\cdots\leftarrow c\to\cdots\to w$
with both $v,w$ appearing exactly once gives directed paths from $c$ to $v$ (not
through $w$, so surviving in $G_{\doit(w)}$) and from $c$ to $w$ (not through $v$).
$\checkmark$

$(\Leftarrow)$: The two directed paths can be concatenated to form a bifurcation.
\textbf{Minor note:} The proof says ``directed path'' but only directed \emph{walks} are
guaranteed.  However, a directed walk can always be shortened to a directed path
(by removing cycles), and the non-passing-through conditions are preserved.
$\checkmark$

\textbf{Verdict: CORRECT.}

\section{Proof 4: Disjoint node-splittings commute (lines 412--438)}
%% ========================================================================

\textbf{Statement (Lemma).}
For disjoint $W_1,W_2\subseteq V$:
$(G_{\spl(W_1)})_{\spl(W_2)} = (G_{\spl(W_2)})_{\spl(W_1)} = G_{\spl(W_1\sqcup W_2)}$.

\textbf{Analysis.}
Component-by-component verification.  The key insight is that since $W_1\cap W_2=\emptyset$,
each $w\in W_2$ is in $V\setminus W_1\subseteq V_{\spl(W_1)}$, so the second
splitting is well-defined.  The superscript tracking ($v^{0_k},v^{1_k}$
for $v\in W_k$) correctly shows that both orderings produce the same edge sets.
$\checkmark$

\textbf{Verdict: CORRECT.}

\section{Proof 5: Disjoint hard interventions and node-splittings commute (lines 448--476)}
%% ========================================================================

\textbf{Statement (Lemma).}
For disjoint $W_1\subseteq J\cup V$, $W_2\subseteq V$:
$(G_{\doit(W_1)})_{\spl(W_2)} = (G_{\spl(W_2)})_{\doit(W_1)}$.

\textbf{Analysis.}
Component-by-component.  The critical observations:
\begin{itemize}

    $v_2^0\in W_1$ iff $v_2\in W_1$.  $\checkmark$
  \item In bidirected edges: $v_k^0\in W_1$ iff $v_k\in W_1$.  $\checkmark$
\end{itemize}

\textbf{Verdict: CORRECT.}

For disjoint $W_1,W_2\subseteq V$:

Similarly for $v_1\in W_2$: $v_1^{i_1}=v_1$, then $(v_1)^{i_2}=v_1^{i_2}$.
$\checkmark$

    Uses $v_k^o\in W_1\iff v_k\in W_1$.  $\checkmark$
\end{itemize}

\textbf{Verdict: CORRECT.}

\section{Proof 13: Marginalizing SWIG output $=$ hard intervention (lines 1078--1111)}
%% ========================================================================

\textbf{Statement (Lemma).}
$G_{\doit(W)} \cong (G_{\swig(W)})^{\setminus W^o}$ via $w\mapsto w^i$.

\textbf{Analysis.}
Key observation: in $G_{\swig(W)}$, nodes $w^o\in W^o$ have no outgoing directed edges
(a tail $w^o$ would require $w^o=v^i$ for some~$v$, impossible by construction).
Therefore marginalization over $W^o$ creates no new edges---the surviving
edges correspond bijectively (via $w^i\mapsto w$) to $E_{\doit(W)}$.
Same for bidirected edges: no bifurcation arm extends through~$W^o$.  $\checkmark$

\textbf{Verdict: CORRECT.}

\section{Proof 14: $\sigma$-separation stable under marginalization (lines 1296--1449)}
%% ========================================================================

\textbf{Statement (Lem.\ \texttt{lem:stability\_separation\_marginalization}).}
$A$ is $i$-$\sigma$-separated from $B$ given $C$ in $G$ iff the same holds in
$G^{\setminus D}$.

\textbf{Analysis.}
This is the most intricate proof in the file.
\begin{enumerate}[leftmargin=*]
  \item Reduces to $D=\{u\}$ by induction on $|D|$.  $\checkmark$
  \item \textbf{Forward direction} (lift walk from $G^{\setminus u}$ to~$G$):
    Each edge in $G^{\setminus u}$ lifts to a short walk through~$u$.
    Arrowhead preservation at non-$u$ nodes verified.
    Inserted $u$-nodes are non-colliders not in~$C$.  $\checkmark$
  \item \textbf{Backward direction} (contract walk in~$G$ to $G^{\setminus u}$):
    Uses Prp.~\texttt{prp:sigma\_opens}(3) to assume all colliders lie in~$C$.
    The subtle case: an unblockable non-collider $b_j$ whose outgoing neighbor
    $b_{j+1}\notin\Sc^{G^{\setminus u}}(b_j)$; a detour through $u$'s SCC restores
    $\sigma$-openness.  All cases verified.  $\checkmark$
\end{enumerate}

\textbf{Verdict: CORRECT.}

\section{Proof 15: Walk replacement in SCC (lines 1497--1516)}
%% ========================================================================

\textbf{Statement (Lem.\ \texttt{lem:replace\_walk}).}
Replacing a subwalk between $v_i,v_j\in\Sc^G(v_j)$ with a shortest directed
path within the SCC preserves $\sigma$-openness.

\textbf{Analysis.}
Case~(i) (forward path): intermediate nodes are unblockable non-colliders.  $\checkmark$

Case~(ii) (backward path): if both endpoints have incoming arrowheads, a collider
must exist between positions $i$ and $j$, placing $v_i\in\Anc^G(C)$; after
replacement $v_i$ becomes a $\sigma$-open collider.  $\checkmark$

\textbf{Verdict: CORRECT.}

\section{Proof 19: $i$-$\sigma$-sep iff $\sigma$-sep in $\bar G$ (lines 2323--2338)}
%% ========================================================================

\textbf{Statement (Lem.\ \texttt{lem:isep\_equiv\_sep}).}
$A\stackrel{\sigma}{\perp}_G B\mid C \;\iff\;
 A\stackrel{\sigma}{\perp}_{\bar G} B\cup J\mid C$,
where $\bar G$ adds all $J$--$J$ bidirected edges.

\textbf{Analysis.}
$(\Rightarrow)$: A minimal $\sigma$-open path in~$G$ from $A$ to $B\cup J$ also
exists in $\bar G$ (which has all edges of~$G$); SCCs are unchanged (same directed
edges).  $\checkmark$

$(\Leftarrow)$: A minimal $\sigma$-open path in $\bar G$ cannot use added $J$--$J$
edges by minimality (any $J$-node reached could serve as endpoint, giving a shorter
path).  $\checkmark$

\textbf{Verdict: CORRECT.}

\section{Proof 20: Symmetrized $i$-$\sigma$-sep (lines 2347--2366)}
%% ========================================================================

\textbf{Statement (Lem.\ \texttt{lem:symmetrized\_isep\_equiv\_sep}).}
$(A\stackrel{\sigma}{\perp}_G B\mid C) \lor (B\stackrel{\sigma}{\perp}_G A\mid C)
\;\iff\; A\stackrel{\sigma}{\perp}_{\bar G} B\mid C$.

\textbf{Analysis.}
$(\Leftarrow)$: Uses Right Decomposition and Lem.~\texttt{lem:isep\_equiv\_sep}.  $\checkmark$

$(\Rightarrow)$ (lines 2356--2365): Concatenates two $\sigma$-open paths through
$J$-nodes $j_1,j_2$.

\textbf{GAP (lines 2362--2364):}
The proof writes the combined walk with notation implying $j_1$ and $j_2$ are
non-colliders (forks).  However, depending on the orientation of the last/first
edges of the original paths, $j_1$ or $j_2$ could become \emph{colliders} in the
concatenated walk (if both adjacent edges point inward).  In that case, one would
need $j_1\in\Anc^{\bar G}(C)$ for $\sigma$-openness, which is not verified.
The result is likely still true but the proof does not handle all edge-orientation
cases.

\textbf{Verdict: GAP in the forward direction.}

\section{Proof 21: $i$-$\sigma$-sep with restricted $\bar G$ (lines 2377--2404)}

\textbf{Statement (Lem.\ \texttt{lem:isep\_equiv\_sep\_restricted}).}
Under conditions $A\cap J=\emptyset$ and $J\subseteq B\cup C$:
$A\stackrel{\sigma}{\perp}_G B\mid C \;\iff\;
 A\stackrel{\sigma}{\perp}_{\bar G} B\mid C$.

\textbf{Analysis.}
$(\Rightarrow)$ (lines 2381--2388): A minimal walk from $A$ to $B\cup J$ in~$G$
must end in~$B$ (not $J\setminus B$, since $J\subseteq B\cup C$ and endpoints
$\notin C$).  SCCs in~$V$ are unchanged.  $\checkmark$

$(\Leftarrow)$ (lines 2390--2403):

\textbf{GAP (line 2397):}
The proof claims a $J$-node $j$ appearing as an interior node of a minimal
$\sigma$-open walk ``must be a non-endpoint non-collider that is not unblockable,''
but does not justify the ``not unblockable'' part.
If $j$ were an unblockable non-collider in $C\setminus B$, it would not block the
walk and would not be in~$B$, so the minimality contradiction would not apply.
While this scenario may be impossible for structural reasons
(the walk's structure near~$j$ may force a child outside $\Sc^{\bar G}(j)$),
the proof does not argue this.

\textbf{Verdict: GAP in the backward direction.}
