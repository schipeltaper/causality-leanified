\section{Proof 2: Prp.\ \texttt{prp:graph\_cause\_of} (lines 425--432)}
%% ========================================================================

\textbf{Statement.}
$a\notin\Anc^G(b) \;\iff\; b\perp^{i\sigma}_{G_{\doit(I_a)}} I_a\mid J\setminus\{I_a\}$.

\textbf{Analysis.}
$(\Rightarrow)$: A $\sigma$-open walk from $b$ to $I_a$ has no colliders
(input nodes can't be colliders), no internal $J$-nodes (would block), hence
is a directed walk $I_a\to a\to\cdots\to b$, giving $a\in\Anc(b)$.
$(\Leftarrow)$: A directed walk $I_a\to\cdots\to b$ has all internal
non-colliders in $V$, hence is $\sigma$-open.  $\checkmark$

\textbf{Verdict: CORRECT.}

\section{Proof 8: Prp.\ \texttt{prp:graph\_unconfounded2} (lines 1220--1245)}
%% ========================================================================

\textbf{Statement.}
$a,b$ unconfounded in $G=(G^+)^{\setminus W}$ $\iff$
$a\perp_{(G^+)^{\mathrm{twin}}_{\doit(a',b)}} b'\mid J,J',a',b$.

\textbf{Analysis.}
$(\Rightarrow)$: A $\sigma$-open walk from $a$ to $b'$ must pass through $W$
(only connection between worlds).  $W$ nodes have no parents, forcing a
directed walk pattern that yields a bifurcation, contradicting unconfoundedness.
$(\Leftarrow)$: A bifurcation in $G$ lifts to a $\sigma$-open walk in the
twin graph.  $\checkmark$

\textbf{Minor:} Line~1233 writes ``$G^+(M)$'' where ``$G^+$'' is intended.

\textbf{Verdict: CORRECT.}

\section{Proof 10: Prp.\ \texttt{prp:scm\_rung3\_unconfounded\_acyclic}
(lines 1383--1392)}
%% ========================================================================

\textbf{Statement.} If $b$ is not a rung-3 cause of $a$: rung-3 unconfoundedness
$\iff$ $g_a(x_J,X_W)\perp g^{\doit(a)}_b(x_{a'},x_{J'},X_W)$ for all inputs.

\textbf{Analysis.}
Since $b$ is not a rung-3 cause of $a$, $g^{\doit(b)}_a$ is essentially
constant in $x_b$.  By consistency: $g^{\doit(b)}_a(x_b,x_J,x_W)=g_a(x_J,x_W)$.
Substitute into the rung-3 unconfoundedness definition.  $\checkmark$

\textbf{Verdict: CORRECT.}

\section{Proof 11: Prp.\ \texttt{prp:rung3\_unconfounded\_implies\_no\_confounding\_bias}
(lines 1484--1510)}
%% ========================================================================

\textbf{Statement.} Rung-3 unconfoundedness $+$ $b$ not rung-3 cause of $a$
$\Rightarrow$ no confounding bias.

\textbf{Analysis.}
\begin{enumerate}[leftmargin=*]
  \item Prp.~10 gives independence $g_a(x_J,X_W)\perp g^{\doit(a)}_b(x_a,x_J,X_W)$.
  \item Consistency: $g_b(x_J,x_W)=g^{\doit(a)}_b(g_a(x_J,x_W),x_J,x_W)$.
  \item Conditioning on $g_a(x_J,X_W)=x_a$:
    $P(g_b\mid g_a=x_a) = P(g^{\doit(a)}_b(x_a,x_J,X_W)\mid g_a=x_a)
    = P(g^{\doit(a)}_b(x_a,x_J,X_W))$
    (by independence).
  \item This is exactly $P_M(X_b\mid\doit(X_J),X_a)=P_M(X_b\mid\doit(X_J),\doit(X_a))$.
    $\checkmark$
\end{enumerate}

\textbf{Verdict: CORRECT.}
