\section{Proof 1: Prp.\ \texttt{prp:G\_test\_asymptotics\_H0} (lines 141--160)}

\textbf{Statement.} Under $H_0:X\perp Y$ and positivity,
$G_N\leadsto\chi^2_\nu$ with $\nu=(|\mathcal{X}|-1)(|\mathcal{Y}|-1)$.

\textbf{Analysis.}
Standard application of Wilks' theorem.  Degree-of-freedom computation:
$\dim(\mathcal{X}_\Theta)-\dim(\mathcal{X}_\Theta^0)
= (|\mathcal{X}||\mathcal{Y}|-1) - (|\mathcal{X}|-1) - (|\mathcal{Y}|-1)
= (|\mathcal{X}|-1)(|\mathcal{Y}|-1)$.  $\checkmark$

\textbf{Verdict: CORRECT.}

\section{Proof 2: Cor.\ \texttt{cor:G\_test\_consistency} (lines 209--264)}

\textbf{Statement.} The $G$-test with thresholds $\tau_N\to\infty$,
$\tau_N/N\to 0$ is asymptotically consistent.

\textbf{Analysis.}
Type~II: SLLN $\Rightarrow$ $G_N/N\to 2I(\theta)>0$ a.s., while $\tau_N/N\to 0$.
$\checkmark$

Type~I: $G_N\leadsto\chi^2_\nu$ (Proof~1); Polya's theorem gives uniform
convergence of CDFs; $\tau_N\to\infty$ gives $F_{G_N}(\tau_N)\to 1$,
hence $P(G_N\ge\tau_N)\to 0$.  $\checkmark$

\textbf{Minor:} Lines 219--220: index notation $\hat\theta_{+x}$,

(per convention at line~35).

\textbf{Verdict: CORRECT.}

\section{Proof 5: Lem.\ \texttt{lem:g\_derivatives\_H0} (lines 266--307)}

\textbf{Statement.} Under $H_0$, the function $g(\theta,\gamma)$ has a
second-order Taylor expansion with zero value, zero gradient, and explicit
Hessian (Kronecker product form).

\textbf{Analysis.}
\begin{itemize}
  \item Zero-order: $\mathrm{KL}(\theta_{X|\ast}\|\theta_{X|\ast})=0$.  $\checkmark$
  \item Gradient: $\log(\theta_{x|c}/\sum_{c'}\gamma_{c'}\theta_{x|c'})=\log 1=0$
    under $H_0$.  $\checkmark$
  \item Hessian: Kronecker product structure verified by direct computation.
    $\checkmark$
  \item Remainder: third derivatives bounded uniformly in $\gamma$ using
    positivity.  $\checkmark$
\end{itemize}

\textbf{Verdict: CORRECT.}

\section{Proof 6: Prp.\ \texttt{prp:ext\_G\_test\_asymptotics\_H0}
(lines 317--403)}

\textbf{Statement.} Under $H_0:X\perp C$ with positive $\theta$ and
$N_{+|c}\to\infty$, the extended $G$-statistic satisfies
$G_N\leadsto\chi^2_{(|\mathcal{X}|-1)(|\mathcal{C}|-1)}$.

\textbf{Analysis.}
The proof uses a Taylor expansion (Lem.~5), a CLT for $S_N$, and a rotational
symmetry argument (Lem.~7) to reduce to a chi-squared limit.

\textbf{GAP 1:} Lem.~7 requires full rotational symmetry of the limiting
distribution, but the actual limit $\mathcal{N}(0,\tilde\Sigma)$ has only
partial rotational symmetry (invariant under $I_\mathcal{X}\otimes U$,
not all rotations).  The argument is fixable by generalizing the lemma,
but as written there is a mismatch.

\textbf{GAP 2:} The Slutsky step $N\cdot o_P(\|\hat\theta-\theta\|^2)\to 0$
requires $N\|\hat\theta-\theta\|^2=O_P(1)$.  Under only $N_{+|c}\to\infty$
(not $N_{+|c}/N\ge\epsilon$), this is not guaranteed.

\textbf{Minor:} Line~336 parenthetical incorrectly describes the Kronecker
vector as ``the all-ones vector $1_\mathcal{X}$''; the actual vector is
$1/\sqrt{\theta_{X|\ast}}$.

\textbf{Verdict: GAP.}

\section{Proof 7: Lem.\ \texttt{lem:weak\_convergence\_to\_rotationally\_symmetric\ldots}
(lines 417--433)}

\textbf{Statement.} If $Q$ is rotationally symmetric on $\mathbb{R}^k$ and
$P_n\leadsto Q$, then $P_n\circ U_n\leadsto Q$ for any rotations $U_n$.

\textbf{Analysis.}
Uses L\'evy--Prokhorov metric + rotation invariance of the metric
+ rotational symmetry of~$Q$.  $\checkmark$

\textbf{Minor typo (line~425):} $SO(\mathbb{R}^n)$ should be $SO(\mathbb{R}^k)$.

\textbf{Verdict: CORRECT.}

\section{Proof 8: Cor.\ \texttt{cor:ext\_G\_test\_consistency}
(lines 451--477)}

\textbf{Statement.} Under positivity and $N_{+|c}/N\ge\epsilon>0$,
the extended $G$-test with thresholds $\tau_N\to\infty$, $\tau_N/N\to 0$

\textbf{Analysis.}

``$\to\infty$''.

\textbf{Verdict: CORRECT.}
