\section{Proof 29: Prp.\ \texttt{prp:Bounding\_counterfactuals}
(lines 1712--1758)}
\textbf{Statement.} Natural bounds on counterfactual probability
$P_{M^{\mathrm{twin}}}((Y')^{\doit(x'=1)}{=}1\mid Y^{\doit(x=0)}{=}0)$
for binary $X,Y$.

\textbf{Analysis.} By Cor.~27, WLOG $W\in\{f_{00},f_{01},f_{10},f_{11}\}$
parameterized by $\rho\in\Delta_3$.  The counterfactual probability equals
$\rho_{01}/q_{0|0}$.  Upper bound: $\rho_{01}\le\min(q_{0|0},q_{1|1})$.
Lower bound: $\rho_{01}\ge q_{0|0}-\min(q_{0|0},q_{0|1})$.  Both derived
from non-negativity of the $\rho$'s.  $\checkmark$

\textbf{Note:} The proof implicitly assumes $q_{0|0}>0$ (needed for conditioning).

\textbf{Verdict: CORRECT.}
