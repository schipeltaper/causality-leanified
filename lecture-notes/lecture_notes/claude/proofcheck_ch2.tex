\section{Proof 1: Lemma \texttt{transmeasure\_is\_twoargfunction} (lines 331--336)}
%% ========================================================================

\textbf{Statement.} One-to-one correspondence between finite transition measures
$K(W|T): \Tcal \to \Msf(\Wcal)$ and two-argument
functions $\tilde{K}: \Bcal_{\Wcal} \times \Tcal \to \R_{\ge 0}$
satisfying (i) measure in first argument and (ii) measurability in second argument.

\textbf{Proof given.} ``The correspondence is via putting
$K(W \in D|T=t)=\tilde K(W \in D|T=t)$ and vice versa.
The corresponding properties hold by definition of the $\sigma$-algebra on
$\Msf(\Wcal)$, also see Remark~\ref{rem:prob-sigma-algebra}.
Working out the details is left as an exercise.''

\textbf{Analysis.}
\begin{itemize}
  \item The proof is left as an exercise.  The claimed correspondence is standard
        and follows directly from Remark~\ref{rem:prob-sigma-algebra} (line~186),
        which states that a map $K: \Xcal \to \Msf(\Ycal)$
        is measurable iff for every $B \in \Bcal_{\Ycal}$ the composition
        $\mathrm{ev}_B \circ K$ is measurable---which is exactly condition~(ii).
        Condition~(i) holds because each $K(t) \in \Msf(\Wcal)$
        is a finite measure by definition.
  \item No issues found.
\end{itemize}

\textbf{Verdict: CORRECT (exercise, but straightforward).}

\section{Proof 2: Lemma \texttt{lem:density-abs-cont} (lines 473--475)}
%% ========================================================================

\textbf{Statement.} If $Q(W|T)$ has a Doob-Radon-Nikodym derivative w.r.t.\
$K(W|T)$ then $Q(W|T) \ll K(W|T)$.

\textbf{Proof given.} ``Should be clear, left as an exercise.''

\textbf{Analysis.}
\begin{itemize}
  \item Indeed straightforward: if $g$ is the Doob-RN derivative, then for any $D$
        with $K(W \in D|T=t)=0$ we have
        $Q(W \in D|T=t) = \int \mathbf{1}_D(w) \cdot g(w|t)\, K(W \in dw|T=t) = 0$,
        since the integrating measure gives zero mass to~$D$.
  \item No issues found.
\end{itemize}

\textbf{Verdict: CORRECT (exercise, but straightforward).}

\section{Proof 3: Corollary \texttt{cor:doob-radon-nikodym} (lines 538--558)}
%% ========================================================================

\textbf{Statement.} Let $\Wcal$ be standard, $P(W|T):\Tcal\to\Msf(\Wcal)$ a finite
transition measure, and $\mu$ a $\sigma$-finite measure on $\Wcal$.  Then
$P(W|T) \ll \mu$ iff $P(W|T)$ has a Doob-RN derivative w.r.t.\ $\mu$, and the
derivative is essentially unique.

\textbf{Line-by-line analysis.}

\begin{enumerate}[leftmargin=*]
  \item \textbf{Lines 539--545: Construction of $Q$ with $Q \ll \mu \ll Q$.}
    \begin{itemize}
      \item \emph{Finite case} ($\mu(\Wcal)<\infty$): $Q(D):=\mu(D)/\mu(\Wcal)$.
            Clearly $Q \ll \mu \ll Q$ (they differ by a positive constant).
            \textbf{Minor issue:} If $\mu = 0$ (the zero measure, which is
            $\sigma$-finite and finite), then $\mu(\Wcal)=0$ and the formula
            involves $0/0$.  However, $P \ll 0$ forces $P=0$, and $p=0$ is a
            trivial Doob-RN derivative.  So the result still holds; only the
            proof construction breaks.  This edge case is harmless in practice.
      \item \emph{$\sigma$-finite, non-finite case}:
            $Q(D) := \sum_{n\in\N} 2^{-n}\,\mu(D\cap\Wcal_n)/\mu(\Wcal_n)$.
            \begin{itemize}
              \item $Q \ll \mu$: if $\mu(D)=0$ then each $\mu(D\cap\Wcal_n)=0$
                    so $Q(D)=0$.  $\checkmark$
              \item $\mu \ll Q$: if $Q(D)=0$ then every term vanishes, so
                    $\mu(D\cap\Wcal_n)=0$ for all~$n$, hence
                    $\mu(D)=\sum_n \mu(D\cap\Wcal_n)=0$.  $\checkmark$
            \end{itemize}
      \item \textbf{Normalization question.}
            The proof claims $Q$ is a \emph{probability measure}.
            $Q(\Wcal)=\sum_{n\in\N} 2^{-n}$.  If $\N=\{1,2,3,\dots\}$ this
            equals~$1$; if $\N=\{0,1,2,\dots\}$ this equals~$2$.  The convention
            for~$\N$ is not stated in the text.  This does \emph{not} affect the
            mathematical argument: only the mutual absolute continuity
            $Q \ll \mu \ll Q$ and finiteness of~$Q$ are used, not
            $Q(\Wcal)=1$.  But strictly speaking, the claim ``probability
            measure'' may be incorrect depending on convention.
      \item The disjoint decomposition $\Wcal = \bigsqcup_{n} \Wcal_n$ with
            $0 < \mu(\Wcal_n) < \infty$ is obtained from a standard
            $\sigma$-finite covering by making it disjoint and discarding
            zero-measure pieces (the discarded union has $\mu$-measure zero).
            $\checkmark$
    \end{itemize}

  \item \textbf{Line 546: RN derivative $q$ of $Q$ w.r.t.\ $\mu$.}
    Applies the standard Radon-Nikodym theorem (Thm.\ \texttt{thm:radon-nikodym},
    line~602 in \texttt{probability\_theory.tex}) with $Q \ll \mu$ and $\mu$
    $\sigma$-finite.  $\checkmark$

    \textbf{TODO:} Verify external reference \cite{Kle20} Cor.\ 7.34.

  \item \textbf{Lines 547--549: Transitivity of absolute continuity.}
    From $P(W|T) \ll \mu$ and $\mu \ll Q(W)$ one gets $P(W|T) \ll Q(W|T)$
    (viewing $Q$ as a constant transition measure):
    $Q(D)=0 \Rightarrow \mu(D)=0 \Rightarrow P(D|t)=0$ for all $t$.  $\checkmark$

  \item \textbf{Line 550: Application of Theorem \texttt{thm:doob-radon-nikodym}.}
    Both $P(W|T)$ and $Q(W|T)$ are finite transition measures,
    $P \ll Q$, and $\Wcal$ is standard.
    Theorem applies and yields Doob-RN derivative $k$ of $P$ w.r.t.\ $Q$.
    $\checkmark$

  \item \textbf{Lines 551--554: Constructing $p(w|t) := k(w|t)\cdot q(w)$.}
    \begin{itemize}
      \item Joint measurability: $k$ is $(\Bcal_\Wcal\otimes\Bcal_\Tcal)$-measurable
            (Doob-RN), $q$ is $\Bcal_\Wcal$-measurable (standard RN), so their
            product is jointly measurable.  $\checkmark$
      \item Verification:
            $P(D|t) = \int_D k(w|t)\,Q(dw) = \int_D k(w|t)\,q(w)\,\mu(dw)$,
            where the second equality uses $Q(dw) = q(w)\,\mu(dw)$.  $\checkmark$
    \end{itemize}

  \item \textbf{Line 557: Reverse direction and essential uniqueness.}
    \begin{itemize}
      \item Reverse ($p$ exists $\Rightarrow$ $P \ll \mu$): immediate from
            Lemma~\texttt{lem:density-abs-cont}.  $\checkmark$
      \item Uniqueness: if $p_1,p_2$ are two such densities, then for all $D$
            and $t$: $\int_D(p_1-p_2)\,d\mu = 0$, so $\{p_1 \neq p_2\}_t$ is
            $\mu$-null for each~$t$.  Standard, parallel to
            Lemma~\texttt{lem:ess-uniq-drnd}.  $\checkmark$
    \end{itemize}
\end{enumerate}

\textbf{Verdict: CORRECT.}  Minor cosmetic issue with the normalization of $Q$
depending on the convention for $\N$.

\section{Proof 4: Corollary \texttt{cor:abs-cont-str-pos-den} (lines 573--603)}
%% ========================================================================

\textbf{Statement.}
\begin{enumerate}
  \item $Q(W|T)$ has a strictly positive Doob-RN derivative w.r.t.\ $K(W|T)$ iff
        $Q \ll K \ll Q$.
  \item $P(W|T)$ has a strictly positive Doob-RN derivative w.r.t.\ $\mu$ iff
        $\mu \ll P \ll \mu$.
\end{enumerate}

\textbf{Line-by-line analysis.}

\begin{enumerate}[leftmargin=*]
  \item \textbf{Line 574: ``The second case follows from the first using the arguments
        from Corollary~\texttt{cor:doob-radon-nikodym}.''}

    This is terse but correct.  The argument:
    \begin{itemize}
      \item[$\Rightarrow$:] If $p>0$ is a density of $P$ w.r.t.\ $\mu$, then
            $P \ll \mu$ (Lemma~\texttt{lem:density-abs-cont}) and
            $\mu \ll P$ (since $P(D|t) = \int_D p\,d\mu = 0$ with $p>0$
            forces $\mu(D)=0$).
      \item[$\Leftarrow$:] Construct $Q$ with $Q \ll \mu \ll Q$ as in
            Corollary~\texttt{cor:doob-radon-nikodym}.  Then
            $Q \ll P \ll Q$ by transitivity, so by Statement~1 there is a
            strictly positive Doob-RN derivative $k>0$ of $P$ w.r.t.\ $Q$.
            Since $Q \ll \mu \ll Q$, the standard RN density $q$ of $Q$ w.r.t.\
            $\mu$ can be chosen strictly positive (modify on a $\mu$-null set
            where $q=0$).  Then $p := k\cdot q > 0$ is a strictly positive
            density of $P$ w.r.t.\ $\mu$.
    \end{itemize}
    No gap, but the argument is somewhat non-trivial to reconstruct from the
    one-line claim.

  \item \textbf{Lines 575--581: $\Rightarrow$ direction of Statement~1.}

    Given: $q > 0$ is a Doob-RN derivative of $Q$ w.r.t.\ $K$.
    \begin{itemize}
      \item $Q \ll K$ by Lemma~\texttt{lem:density-abs-cont}.  $\checkmark$
      \item Define $k(w|t) := 1/q(w|t) > 0$.  Joint measurability: $q$ is
            jointly measurable and strictly positive, so $1/q$ is jointly
            measurable (composition with the continuous map $r\mapsto 1/r$
            on $(0,\infty)$).  $\checkmark$
      \item Verification that $k$ is a density of $K$ w.r.t.\ $Q$:
            \[ \int_D k\,dQ_t = \int_D \frac{1}{q(w|t)}\,q(w|t)\,K(dw|t)
                              = K(D|t). \quad\checkmark \]
      \item $K \ll Q$ by Lemma~\texttt{lem:density-abs-cont}.  $\checkmark$
    \end{itemize}

  \item \textbf{Lines 583--603: $\Leftarrow$ direction of Statement~1.}

    Given: $Q \ll K \ll Q$.  Apply Theorem~\texttt{thm:doob-radon-nikodym}
    (requires $\Wcal$ standard, which is assumed) to get Doob-RN derivatives:
    $q$ of $Q$ w.r.t.\ $K$ and $k$ of $K$ w.r.t.\ $Q$.

    \begin{itemize}
      \item \textbf{Lines 587--591:} Compute
        $\int_D 1\,Q(dw|t) = \int_D q\cdot k\,Q(dw|t)$ for all $D,t$.

        The chain: $Q(D|t) = \int_D q\,K(dw|t) = \int_D q\cdot k\,Q(dw|t)$,
        where the second step substitutes $K(dw|t) = k(w|t)\,Q(dw|t)$
        (extending the Doob-RN identity from indicators to non-negative
        measurable functions via monotone convergence).  $\checkmark$

      \item \textbf{Lines 592--594:} The set $N := \{q\cdot k \neq 1\}$ is a
        $Q$-null set (by the same argument as
        Lemma~\texttt{lem:doob-radon-nikodym-unique}: $\int_D(1-qk)\,dQ_t=0$
        for all $D$ implies $\{qk>1\}$ and $\{qk<1\}$ are both $Q$-null).
        Since $K \ll Q$, also a $K$-null set.  $\checkmark$

        Measurability of $N$: $q\cdot k$ is jointly measurable, so
        $N = (q\cdot k)^{-1}(\R\setminus\{1\}) \in \Bcal_\Wcal\otimes\Bcal_\Tcal$.
        $\checkmark$

      \item \textbf{Lines 596--599:} Define modified densities
        $\tilde{q} := q\cdot\mathbf{1}_{N^c} + \mathbf{1}_N$ and
        $\tilde{k} := k\cdot\mathbf{1}_{N^c} + \mathbf{1}_N$.

        Still Doob-RN derivatives: e.g.,
        $\int_D\tilde{q}\,K(dw|t)
          = \int_D q\,K(dw|t) - \underbrace{\int_{D\cap N_t} q\,K(dw|t)}_{=0}
            + \underbrace{\int_{D\cap N_t} 1\,K(dw|t)}_{=0} = Q(D|t)$,
        since $N$ is a $K$-null set. Similarly for $\tilde{k}$.  $\checkmark$

      \item \textbf{Lines 600--602:} $\tilde{q}\cdot\tilde{k} = 1$ everywhere.
        On $N^c$: $q\cdot k = 1$ by definition of~$N$.
        On $N$: $1\cdot 1 = 1$.
        Since $\tilde{q},\tilde{k}\ge 0$ and their product is~$1$,
        both must be strictly positive.  $\checkmark$
    \end{itemize}
\end{enumerate}

\textbf{Verdict: CORRECT.}

\section{Proof 5: Lemma \texttt{lem:doob-radon-nikodym-unique} (lines 617--633)}
%% ========================================================================

\textbf{Statement.} If $g_1,g_2$ are two Doob-RN derivatives of $Q(W|T)$
w.r.t.\ $K(W|T)$, then $N:=\{g_1 \neq g_2\}$ is a $K(W|T)$-null set and is
in $\Bcal_\Wcal\otimes\Bcal_\Tcal$.

\textbf{Line-by-line analysis.}

\begin{enumerate}[leftmargin=*]
  \item \textbf{Lines 618--622:} Define
    $N^> := \{g_1 > g_2\} = (g_1,g_2)^{-1}(\Delta^>)$ where
    $\Delta^>=\{(r_1,r_2)\in\R^2:r_1>r_2\}$ is an open (hence Borel) set.
    Since $g_1,g_2$ are jointly measurable,
    $N^> \in \Bcal_\Wcal\otimes\Bcal_\Tcal$.  $\checkmark$

  \item \textbf{Line 623:} $N^>_t := \{w:(w,t)\in N^>\}\in\Bcal_\Wcal$.
    Sections of product-measurable sets are measurable.  $\checkmark$

  \item \textbf{Lines 625--629:} The computation
    \begin{align*}
      0 &= Q(N^>_t|t) - Q(N^>_t|t) \\
        &= \int \mathbf{1}_{N^>_t}\cdot g_1\,K(dw|t)
         - \int \mathbf{1}_{N^>_t}\cdot g_2\,K(dw|t) \\
        &= \int \underbrace{\mathbf{1}_{N^>_t}\cdot(g_1-g_2)}_{%
          \ge 0 \text{ everywhere, } > 0 \text{ on } N^>_t} K(dw|t).
    \end{align*}
    The second line uses the Doob-RN definition applied to $D=N^>_t$.
    The integrand is non-negative everywhere (zero outside $N^>_t$, positive
    on $N^>_t$).  A non-negative function integrating to zero implies the set
    where it is positive has measure zero: $K(N^>_t|t)=0$.  $\checkmark$

  \item \textbf{Lines 630--632:} By symmetry (swapping $g_1,g_2$),
    $K(N^<_t|t)=0$.  Then $N = N^>\cup N^<$ and
    $K(N_t|t) \le K(N^>_t|t)+K(N^<_t|t) = 0$.  $\checkmark$
\end{enumerate}

\textbf{Verdict: CORRECT.}  Clean, complete, no gaps.

\section{Proof 6: Existence of the Doob-RN derivative (lines 643--719)}
%% ========================================================================

\textbf{Statement.} Let $\Wcal$ be a standard measurable space,
$Q(W|T)\ll K(W|T)$ finite transition measures.  Then $Q$ has a Doob-RN
derivative w.r.t.\ $K$.  (This is the existence direction of
Theorem~\texttt{thm:doob-radon-nikodym}.)

This is labeled a \textbf{proof sketch}.

\textbf{Line-by-line analysis.}

\begin{enumerate}[leftmargin=*]
  \item \textbf{Lines 644--653: Setup of partitions.}
    \begin{itemize}
      \item $\Bcal_\Wcal=\sigma(\Scal)$ with countable
            $\Scal=\{D_n:n\in\N\}$: correct, this is the definition of
            ``countably generated,'' which follows from $\Wcal$ being standard.
            $\checkmark$
      \item The inductive partition construction $\Ecal_0=\{\Wcal\}$,
            $\Ecal_{n+1} = (\bigcup_{D\in\Ecal_n}\{D\setminus D_n, D\cap D_n\})
            \setminus\{\emptyset\}$ produces nested finite partitions that
            generate $\Bcal_\Wcal$.  $\checkmark$
      \item $\Bcal_n \subseteq \Bcal_{n+1} \subseteq \Bcal_\Wcal =
            \sigma(\bigcup_m\Ecal_m)$.  $\checkmark$
    \end{itemize}

  \item \textbf{Lines 654--663: Definition of $q_D$ and measurability.}
    $q_D(t) = Q(D|t)/K(D|t)\cdot\mathbf{1}_{K(D|t)>0}$.
    Since both $t\mapsto Q(D|t)$ and $t\mapsto K(D|t)$ are
    $\Bcal_\Tcal$-measurable (by the transition measure property),
    $q_D$ is $\Bcal_\Tcal$-measurable (ratio on the open set $\{K(D|t)>0\}$,
    zero elsewhere).  $\checkmark$

  \item \textbf{Lines 664--670: Definition of $G_n$, $G$, $g$ and joint measurability.}
    $G_n(w,t) = \sum_{D\in\Ecal_n}\mathbf{1}_D(w)\cdot q_D(t)$: finite sum
    of products of measurable functions, so jointly measurable.  $\checkmark$

    $G = \liminf_n G_n$: countable limit inferior of measurable functions,
    hence measurable.  $\checkmark$

    $g = G\cdot\mathbf{1}_{G<\infty}$: measurable.  $\checkmark$

    \textbf{Note:} Using $\liminf$ (rather than $\lim$) is important for
    defining $G$ everywhere (not just almost everywhere), ensuring joint
    measurability---this is the key technique of the Doob approach.

  \item \textbf{Lines 681--694: The big computation $G_n^t = \E_t[G_m^t|\Bcal_n]$.}

    Checking each step:
    \begin{itemize}
      \item \emph{Line 685 $\to$ 686 (splitting $Q(D|t)$ over finer partition):}
        $Q(D|t) = \sum_{A\in\Ecal_m,\,A\subseteq D} Q(A|t)$ by finite additivity
        of $Q(W|T=t)$ and the disjoint decomposition
        $D = \bigsqcup_{A\subseteq D} A$.  $\checkmark$

      \item \emph{Line 686 $\to$ 687 (inserting $\mathbf{1}_{Q(A|t)>0}$):}
        Terms with $Q(A|t)=0$ contribute nothing, so inserting the indicator
        doesn't change the sum.  $\checkmark$

      \item \emph{Line 687 $\to$ 688 (replacing $\mathbf{1}_{Q>0}$ by
        $\mathbf{1}_{K>0}$, using $Q\ll K$):}
        If $Q(A|t)>0$ then $K(A|t)>0$ (by $Q\ll K$: $K(A|t)=0\Rightarrow Q(A|t)=0$).
        If $Q(A|t)=0$ the whole term vanishes regardless.
        So the replacement is valid.  $\checkmark$

      \item \emph{Line 688 $\to$ 689 (splitting fraction):}
        $Q(A|t)/K(D|t) = [Q(A|t)/K(A|t)]\cdot[K(A|t)/K(D|t)]$
        whenever $K(A|t)>0$ (guaranteed by the indicator), and the term is zero
        when $K(A|t)=0$.  $\checkmark$

      \item \emph{Line 689 $\to$ 690--691 (swapping summation order):}
        Valid for finite sums.  For each $A\in\Ecal_m$ there is exactly one
        $D\in\Ecal_n$ with $D\supseteq A$ (since $\Ecal_n$ is a partition
        and $\Ecal_m$ refines it).  $\checkmark$

      \item \emph{Line 691: Identification of the braced expression as
        $\E_t[\mathbf{1}_A|\Bcal_n](w)$.}

        For $w\in D'\in\Ecal_n$, this equals
        $K(A|t)/K(D'|t)\cdot\mathbf{1}_{A\subseteq D'}\cdot\mathbf{1}_{K(D'|t)>0}$,
        which is the standard formula for the conditional expectation of
        $\mathbf{1}_A$ given the finite partition~$\Ecal_n$ w.r.t.\ $K(W|T=t)$.
        $\checkmark$

        \textbf{TODO:} Verify external reference \cite{Kle20} Lem.\ 8.10
        for the precise form of conditional expectation on atoms of measure zero.

      \item \emph{Lines 692--694 (linearity of conditional expectation):}
        The coefficients $q_A(t)$ are constants (not functions of~$w$), so they
        can be pulled inside/outside the conditional expectation.  $\checkmark$
    \end{itemize}

    \textbf{Typographical error on line 691:} There is a spurious
    ``$\cdot$'' (multiplication dot) immediately after the summation symbol
    $\sum_{D\in\Ecal_n,\,D\supseteq A}$ and before the fraction.  It
    should read
    $\sum_{D\supseteq A} \frac{K(A|t)}{K(D|t)}\cdots$
    without the leading dot.

  \item \textbf{Lines 699--700: Conclusion that $(G_n^t)$ is a martingale.}
    $G_n^t = \E_t[G_m^t|\Bcal_n]$ for all $m\ge n$ is the defining property
    of a martingale w.r.t.\ the filtration $(\Bcal_n)$.  $\checkmark$

  \item \textbf{Line 701: Uniform integrability.}
    ``$\ldots$uniformly integrable w.r.t.\ $K(W|T=t)$, see \cite{Kle20} Ex.\ 7.39.''

    This is stated without proof.  The standard justification: for each fixed~$t$,
    the standard (non-Doob) Radon-Nikodym theorem gives $h_t\in L^1(K(W|T=t))$
    with $Q(D|t)=\int_D h_t\,K(dw|t)$.  One can verify that
    $G_n^t = \E_t[h_t|\Bcal_n]$, making $(G_n^t)$ a Doob martingale
    (conditional expectations of an $L^1$ function), which is uniformly
    integrable.  This is a classical result (not circular: the standard RN
    theorem is proved independently of the \emph{Doob} RN theorem).

    \textbf{TODO:} Verify external reference \cite{Kle20} Ex.\ 7.39.

  \item \textbf{Lines 702--703: Martingale convergence.}
    By the convergence theorem for uniformly integrable martingales:
    $G_n^t \to G^t$ a.s.\ and in $L^1$, and
    $G_n^t = \E_t[G^t|\Bcal_n]$ for all~$n$.  $\checkmark$

    \textbf{TODO:} Verify external reference \cite{Kle20} Thm.\ 11.7.

  \item \textbf{Line 703: $\mathbf{1}_D\cdot G_n^t = \E_t[\mathbf{1}_D\cdot G^t|\Bcal_n]$
    for $D\in\Ecal_n$.}
    Since $\mathbf{1}_D$ is $\Bcal_n$-measurable:
    $\E_t[\mathbf{1}_D\cdot G^t|\Bcal_n] = \mathbf{1}_D\cdot\E_t[G^t|\Bcal_n]
    = \mathbf{1}_D\cdot G_n^t$.  $\checkmark$

  \item \textbf{Lines 705--708: Tower property and final identity.}
    \begin{align*}
      \E_t[\mathbf{1}_D\cdot G^t]
        &= \E_t[\E_t[\mathbf{1}_D\cdot G^t|\Bcal_n]]
         = \E_t[\mathbf{1}_D\cdot G_n^t].
    \end{align*}
    And for $D\in\Ecal_n$:
    $\E_t[\mathbf{1}_D\cdot G_n^t] = q_D(t)\cdot K(D|t)$.
    If $K(D|t)>0$: $= Q(D|t)$.  If $K(D|t)=0$: $q_D(t)=0$ and
    $Q(D|t)=0$ (by $Q\ll K$), so $0=0$.  $\checkmark$

  \item \textbf{Lines 710--711: Extension to all $D\in\Bcal_\Wcal$.}
    ``Since this holds for all $D\in\Ecal_n$ and all $n$, it also holds for
    all $D\in\sigma(\bigcup_n\Ecal_n)$.''

    This step is stated without detailed justification.  The rigorous argument
    uses the Dynkin $\pi$-$\lambda$ theorem: the collection
    $\Dcal := \{D\in\Bcal_\Wcal : \int\mathbf{1}_D G^t\,K(dw|t)=Q(D|t)\}$
    is a $\lambda$-system (contains $\Wcal$; closed under complements and
    countable disjoint unions---the latter by monotone convergence, using
    $G^t\ge 0$) and contains the $\pi$-system
    $\bigcup_n\Ecal_n\cup\{\emptyset\}$.  Therefore
    $\Bcal_\Wcal = \sigma(\bigcup_n\Ecal_n) \subseteq \Dcal$.

    The argument is standard but \textbf{omitted from the proof sketch}.
    Not a gap per se (it's labeled a sketch), but the $\pi$-$\lambda$
    step deserves mention.

\section{Proof 36: Lemma \texttt{cdf} (proof-icdf.tex, lines 18--62)}
%% ========================================================================

\textbf{Statement.} Properties of the CDF $F(x):=P([-\infty,x])$ and quantile
function $R(t):=\inf F^{-1}([t,1])$ on $\bar\R$: monotonicity,
(right/left) continuity, Galois connection $t\le F(x)\iff R(t)\le x$,
reflexive generalized inverse identities $F\circ R\circ F=F$ and
$R\circ F\circ R=R$, and the push-forward $R_*\lambda=P$.

\textbf{Analysis.}
\begin{enumerate}[leftmargin=*]
  \item \textbf{Lines 19--21:} $F$ non-decreasing, right-continuous, $F(\infty)=1$,
    countably many discontinuities (injection into~$\Q$).  $\checkmark$
  \item \textbf{Lines 22--26:} $R(t)=\min F^{-1}([t,1])$ by right-continuity:
    take $x_n\downarrow R(t)$ with $F(x_n)\ge t$, then $F(R(t))\ge t$.  $\checkmark$
  \item \textbf{Lines 27--30:} $R$ non-decreasing, left-continuous, $R(0)=-\infty$.
    Left-continuity: for $t_n\uparrow t$, $t\le\sup_n F(R(t_n))\le F(\sup_n R(t_n))$,
    so $\sup_n R(t_n)\ge R(t)$; combined with monotonicity gives equality.  $\checkmark$
  \item \textbf{Lines 31--41:} Galois connection and reflexive inverse properties.
    All follow from $F(R(t))\ge t$ and monotonicity.  $\checkmark$
  \item \textbf{Lines 51--61:} $R_*\lambda=P$:
    $(R_*\lambda)([-\infty,x])=\lambda(\{t:R(t)\le x\})=\lambda([0,F(x)])=F(x)=P([-\infty,x])$.
    Agreement on a generating $\pi$-system implies equality.  $\checkmark$
\end{enumerate}

\textbf{Verdict: CORRECT.}
