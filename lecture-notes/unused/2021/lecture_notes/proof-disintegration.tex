\begin{Rem}[Existence of conditional Markov kernels]
    If $K(X,Y|Z)$ is a Markov kernel then we want $K(X|Y,Z)$ such that:
    \[K(X,Y|Z) = K(X|Y,Z) \otimes K(Y|Z)\]
    holds. The heuristic here is to find something like a Radon-Nikodym derivative:
    \[ K(X \in A|Y=y,Z=z) = \frac{K(X \in A,Y \in dy |Z=z)}{\phantom{X\in A,\,}K(Y \in dy|Z=z)}(y),  \]
    in a way that it is still a probability measure in $X$ and jointly measurable in $(y,z)$.\\
    To achieve measurability from the start we will  make use of Besicovitch density theorem \ref{besicovitch-density}:
    \[ \frac{K(X \in A,Y \in dy |Z=z)}{\phantom{X\in A,\,}K(Y \in dy|Z=z)}(y) = \lim_{\varepsilon \to 0} \frac{K(X \in A,Y \in B_{\varepsilon}(y) |Z=z)}{\phantom{X\in A,\,}K(Y \in B_{\varepsilon}(y) |Z=z)},   \]
    and the fact that limits of countably many measurable functions stay measurable. 
    To make sure we get a probability measure out we need to evaluate on sufficiently many sets $A$, which we will 
    restrict to sets of the form $A=[0,x]$.
    We will adjust the construction above on null-sets to make sure it will be a probability measure everywhere.\\
    We will show for $\Xcal=\Ycal=\R$ (or $\bar \R$ or $[0,1]$, etc.) and $K(Y|Z)$-almost-all $(y,z) \in \Ycal \times \Zcal$ 
    and all $x\in \Xcal$ that we have that:
    \[  K(X \le x|Y=y,Z=z) = \inf_{m \in \N} \limsup_{n \in \N} 
        \frac{K(X \le\lceil x\rceil_m,
    Y \in [y-\frac{1}{n},y+\frac{1}{n}] |Z=z)}{\phantom{X \le\lceil x\rceil_m,\,}K(Y \in [y-\frac{1}{n},y+\frac{1}{n}] |Z=z)},  \]
    where $x < \lceil x\rceil_m:=\frac{\lfloor mx+1\rfloor}{m} \le x + \frac{1}{m}$ for $m \in \N$.
    On the remaining points $(y,z)$ of the $K(Y|Z)$-null set we can take a somewhat arbitrary choice, e.g.\ we can put:\\
    $K(X \le x|Y=y,Z=z) := K(X\le x|Z=z)$.
\end{Rem}

\begin{Thm}[Existence of conditional Markov kernels]
\label{reg-cond-trans-kernel-existence-I}
Let $\Xcal$ and $\Ycal$ be standard measurable spaces and $\Zcal$ any measurable space. Let 
\[K(X,Y|Z): \, \Zcal \dshto  \Xcal \times \Ycal,\]
be a Markov kernel in two variables. 
Then a conditional Markov kernel
conditioned on $Y$ given $Z$:
\[K(X|Y,Z): \, \Ycal \times \Zcal \dshto  \Xcal,\]
satisfying:
\[K(X,Y|Z) = K(X|Y,Z) \otimes K(Y|Z),\]
exists and is essentially unique.
\begin{proof}
    Since $\Xcal$ and $\Ycal$ are standard measurable spaces we can
without loss of generality assume that $\Xcal=\Ycal=[0,1]$.\\
%The strategy of the proof now is to get an expression for the conditional Markov kernel as a Radon-Nikodym derivative.
%The main problem is to get the measurability, not only in $y$ for fixed $z$, but jointly in $(y,z)$.
For $x \in \Xcal$, $y \in \Ycal$ and $z \in \Zcal$ we put:
    \begin{align*}
        F_0(x,y|z) &:= K(X \le x, Y< y|Z=z),\\
        F_1(x,y|z) &:= K(X \le x, Y\le y|Z=z).
    \end{align*}
Then the functions:
    \[ F_0,\,F_1:\, \Xcal \times \Ycal \times \Zcal \to [0,1],\]
are both (jointly) measurable by \ref{ccdf-measurable}.\\
For $n \in \N$  we further want to define the following functions, anticipating the use of the Besocovitch density theorem \ref{besicovitch-density} 
for the closed balls/intervals
$B_{1/n}(y)=[y-\frac{1}{n},y+\frac{1}{n}]$):
\[F_n(x|y,z):=\frac{K(X \le x, Y \in [y-\frac{1}{n},y+\frac{1}{n}]|Z=z)}%
{\phantom{X \le x,\,}K(Y \in [y-\frac{1}{n},y+\frac{1}{n}]|Z=z)},\] 
as long as we do not divide by $0$.
More formally, using the functions from above, we then actually define:
$$\begin{array}{rcrl}
F_n(x|y,z)
&:=&  \frac{F_1(x,y+\frac{1}{n}|z) - F_0(x,y-\frac{1}{n}|z) }{F_1(1,y+\frac{1}{n}|z) - F_0(1,y-\frac{1}{n}|z)} &\cdot \I_{ F_1(1,y+\frac{1}{n}|z) > F_0(1,y-\frac{1}{n}|z)}   \\
&& + F_1(x,1|z) &\cdot \I_{ F_1(1,y+\frac{1}{n}|z) = F_0(1,y-\frac{1}{n}|z) }.
\end{array} $$
Since $F_0,F_1$ from above are measurable so are all the $F_n$, $n \in \N$.
Since measurability is preserved under $\inf_{n \in \N}$ and $\sup_{n \in \N}$ also the following maps are jointly measurable:
$$\begin{array}{rcrl}
\overline  F(x|y,z) &:=&  \limsup_{ n \to \infty} F_n(x|y,z), \\
\underline F(x|y,z) &:=&  \liminf_{ n \to \infty} F_n(x|y,z).
\end{array}$$
It follows that the set $S \ins \Xcal \times \Ycal \times \Zcal$ of all points $(x,y,z)$ where either $\overline  F(x|y,z) \neq \underline  F(x|y,z)$ or $\overline  F(x|y,z) \notin \R_{\ge 0}$ or $\underline  F(x|y,z) \notin \R_{\ge 0}$ lies in 
$\Bcal_\Xcal \otimes \Bcal_\Ycal \otimes \Bcal_\Zcal$.
Now the Besicovitch density theorem \ref{besicovitch-density} shows that for fixed $x \in \Xcal$ and $z \in \Zcal$
the limits from above agree and are finite $K(Y|Z=z)$-almost everywhere. In other words:
$$ K(Y \in S_{(x,z)} | Z=z) =0, $$
with $S_{(x,z)}:=\{ y \in \Ycal \,|\, (x,y,z) \in S \}$. Furthermore, the Besicovitch density theorem \ref{besicovitch-density} says that for every 
$y \in S_{(x,z)}^\cmpl$ we have:
\[\overline  F(x|y,z) = \underline  F(x|y,z) = \frac{K(X \le x, Y \in dy|Z=z)}{\phantom{X \le x,\,}K(Y \in dy|Z=z) }(y), \]
in other words that both left hand sides are versions of the Radon-Nikodym derivative in $y$ (for fixed $x$ and $z$).\\
By the defining property of Radon-Nikodym derivative we get that for fixed $x$ and $z$ and every $B \in \Bcal_\Ycal$ we have the equation:
\[  K(X \le x, Y \in B |Z=z) = \int \I_B(y) \cdot \overline F(x|y,z) \, K(Y \in dy|Z=z).\]
%(plus the joint measurability mentioned above).\\ 
As a next step we want to modify  $\overline F(x|y,z)$ such that it becomes a cumulative distribution function in $x$, i.e.\ it corresponds to a probability distribution on $\Xcal$. 
For this define $\Xcal_\Q:= \Xcal \cap \Q$, which is countable and dense in $\Xcal$.
First note that $F_n(1|y,z)=1$ for all $(y,z)$, which implies that:
$$\overline  F(1|y,z) = \underline  F(1|y,z) = 1 $$
for all $(y,z) \in \Ycal \times \Zcal$. \\
Then define $S':= \bigcup_{x \in \Xcal_\Q} S_x$ with $S_x:=\{(y,z)\,|\, (x,y,z) \in S\}$. 
$$ K(Y \in S'_z | Z=z) = K(Y \in \bigcup_{x \in \Xcal_\Q} S_{(x,z)} | Z=z) = 0 $$
then shows that $S'$ is a $K(Y|Z)$-zero set in $\Bcal_\Ycal \otimes \Bcal_\Zcal$.\\
For every pair $x_1<x_2$ in $\Xcal_\Q$ consider:
$$ E_{(x_1,x_2)} := \{ (y,z) \,|\, \overline F(x_1|y,z) > \overline F(x_2|y,z) \} \in \Bcal_\Ycal \otimes \Bcal_\Zcal. $$ 
Since we have the equations:
$$ \begin{array}{rcl}
 &&\int \I_{E_{(x_1,x_2),z}}(y) \cdot \overline F(x_1|y,z) \, K(Y \in dy|Z=z) \\
&=& K(X \le x_1, Y \in E_{(x_1,x_2),z} |Z=z) \\
&\stackrel{x_1 < x_2}{\le}& K(X \le x_2, Y \in E_{(x_1,x_2),z} |Z=z)  \\
&=& \int \I_{E_{(x_1,x_2),z}}(y) \cdot \overline F(x_2|y,z) \, K(Y \in dy|Z=z) \\
&\stackrel{\overline F(x_2|y,z) < \overline F(x_1|y,z)}{\le} & \int \I_{E_{(x_1,x_2),z}}(y) \cdot \overline F(x_1|y,z) \, K(Y \in dy|Z=z)
\end{array}$$
we necessarily have $K(Y \in E_{(x_1,x_2),z} |Z=z) = 0$ for every $z \in \Zcal$.
Then $E:= \bigcup_{x_1 < x_2 \in \Xcal_\Q} E_{(x_1,x_2)} \cup S'$ is also a $K(Y|Z)$-zero set in $\Bcal_\Ycal \otimes \Bcal_\Zcal$.
So for every $(y,z) \in E^\cmpl$ and every $x_1,x_2 \in \Xcal_\Q$, $x_1 \le x_2$, we get:
$$\underline F(x_1|y,z) = \overline F(x_1|y,z) \le \overline F(x_2|y,z) = \underline F(x_2|y,z). $$
Now for $x \in \Xcal_\Q$ put: 
$$ D_x := \{ (y,z) \,|\, \overline F(x|y,z) <  \inf_{n \in \N} \overline F(x+1/n|y,z)  \} \in \Bcal_\Ycal \otimes \Bcal_\Zcal.$$
By the dominated convergence theorem (see \cite{Kle14}  Cor. 6.26) we get:
$$ \begin{array}{rcl}
&&  \int \I_{D_{x,z}}(y) \cdot \overline F(x|y,z) \, K(Y \in dy|Z=z) \\
&\stackrel{D_x}{\le}& \int \I_{D_{x,z}}(y) \cdot \inf_{n \in \N} \overline F(x+\frac{1}{n}|y,z) \, K(Y \in dy|Z=z) \\
&=& \inf_{n \in \N} \int \I_{D_{x,z}}(y) \cdot  \overline F(x+\frac{1}{n}|y,z) \, K(Y \in dy|Z=z) \\
&=& \inf_{n \in \N} K(X \le x + \frac{1}{n}, Y \in D_{x,z} |Z=z) \\
&=&  K(X \le x, Y \in D_{x,z} |Z=z) \\
&=& \int \I_{D_{x,z}}(y) \cdot \overline F(x|y,z) \, K(Y \in dy|Z=z).
\end{array}$$
This shows that $K(Y \in D_{x,z}|Z=z)=0$ for all $z \in \Zcal$.
So $D:= \bigcup_{x \in \Xcal_\Q} D_x \cup E$ is again a $K(Y|Z)$-zero set in $\Bcal_\Ycal \otimes \Bcal_\Zcal$.\\
So for all 
%$x \in \Xcal_\Q$ and $(y,z) \in D^\cmpl$ 
$(x,y,z) \in \Xcal_\Q \times D^\cmpl$
the following is then well-defined:
\[ F_+(x|y,z) := \overline F(x|y,z) = \underline F(x|y,z).\]
When restricted to $\Xcal_\Q \times D^\cmpl$ the function $F_+$ is thus 
well-defined, jointly measurable and monotone non-decreasing and continuous from above in $x$ and with $F_+(1|x,z)=1$.
We now aim to extend $F_+$ to $\Xcal \times \Ycal \times \Zcal$.\\
For $x \in \Xcal =[0,1]$ and $n \in \N$ put $\lceil x\rceil_n:=\min(1,\lfloor nx+1\rfloor/n)$. 
Then $\lceil x\rceil_n \in [0,1] \cap \Q = \Xcal_\Q$. 
The map $x \mapsto \lceil x\rceil_n$ is measurable and for $x \in [0,1)$ we have: 
$$x < \lceil x\rceil_n \le x + \frac{1}{n}.$$ 
So $\lceil 1\rceil_n=1$ and  $\lceil x\rceil_n \in \Xcal_\Q$ converges to $x \in \Xcal$, $x \neq 1$, from above for $n \to \infty$.\\
We then define for all $(x,y,z) \in \Xcal \times \Ycal \times \Zcal$: 
$$ F_+(x|y,z) := \inf_{n \in \N} \left \{ \overline F(\lceil x\rceil_n|y,z)\right\} \cdot \I_{D^\cmpl}(y,z) + F_1(x,1|z) \cdot \I_D(y,z). $$ 
It is clear that $F_+$ is again jointly measurable and agrees with the $F_+$, $\overline F$ and $\underline F$ from before on $\Xcal_\Q \times D^\cmpl$ by construction.
As a monotone approximation from above it is clearly continuous from above, monotone non-decreasing and satisfies $F_+(1|y,z)=1$ for all $(y,z)$.
%agrees with $\overline F$ in all (right-)continuity points in $x$.
So for fixed $(y,z)$ now $F_+(\cdot|y,z)$ corresponds to a probability distribution $K(X|Y=y,Z=z)$ on $\Bcal_\Xcal$, uniquely given by the defining relations on sets $[0,x]$:
$$F_+(x|y,z) =: K(X \le x|Y=y,Z=z)$$
for all $x \in \Xcal$. We now show that $(y,z) \mapsto K(X \in A|Y=y,Z=z)$ is measurable for all $A \in \Bcal_\Xcal$.
For this define:
$$ \Dcal:= \{ A \in \Bcal_\Xcal \,|\, (y,z) \mapsto K(X \in A|Y=y,Z=z) \text{ is } (\Bcal_\Ycal\otimes\Bcal_\Zcal)\text{-}\Bcal_\R\text{-measurable} \}. $$
Since $K(X|Y=y,Z=z)$ is a probability measure in $X$ the system $\Dcal$ is closed under countable disjoint unions and complements and contains $\Xcal=[0,1]$.
So $\Dcal$ is a Dynkin system.  
Since the map $(y,z) \mapsto K(X \le x|Y=y,Z=z)=F_+(x|y,z)$ is measurable in $(y,z)$ for all $x \in \Xcal_\Q$ we have 
$\Ecal:=\{ [0,x] \,|\, x \in \Xcal_\Q\} \ins \Dcal$. Since $\Ecal$ is closed under finite intersections 
Dynkin's lemma (see \cite{Kle14} Thm. 1.19) gives $\Bcal_\Xcal=\sigma(\Ecal) \ins \Dcal$ and $K(X|Y,Z)$ is shown to be measurable in $(y,z)$ for all $A \in \Bcal_\Xcal$. So $K(X|Y,Z)$ is a Markov kernel.\\
Finally, we need to show that for all $A \in \Bcal_\Xcal$, $B \in \Bcal_\Ycal$ and $z \in \Zcal$:
$$ K(X \in A, Y \in B |Z=z) = \int \I_B(y) \cdot K(X \in A | Y=y,Z=z) \, K(Y \in dy | Z=z) $$ 
holds. 
Since for any $B \in \Bcal_\Ycal$ and $A=[0,x]$ for $x \in \Xcal_\Q$ we have:
$$\I_B(y) \cdot K(X \le x | Y=y,Z=z) = \I_B(y) \cdot \overline F(x|y,z)$$
up to the $K(Y|Z=z)$-zero set $D_z$ we already have aboves equation on a generating set that is stable under finite intersections.
Again Dynkin's lemma (see \cite{Kle14} Thm. 1.19) shows that the equality from above holds for all sets $A \in \Bcal_\Xcal$, $B \in \Bcal_\Ycal$ and every $z \in \Zcal$.
This concludes the proof.
For the essential uniqueness see \ref{ess-unique}.
\end{proof}
\end{Thm}

\begin{Lem}[Essential uniqueness]
    \label{ess-unique}
    If we have Markov kernels:    
    \[P(X,Y,Z),\,Q(X|Y,Z):\, \Ycal\times \Zcal \dshto \Xcal,\]
    and 
    \[K(Y|Z):\,\Zcal \dshto \Ycal\]
    with any measurable spaces $\Xcal$, $\Ycal$, $\Zcal$ such that:
    \[ P(X|Y,Z) \otimes K(Y|Z) = Q(X|Y,Z) \otimes K(Y|Z), \]
    then for every $A \in \Bcal_\Xcal$ the set:
    \[N_A := \lC (y,z) \in \Ycal \times \Zcal \,|\, P(X \in A|Y=y,Z=z) \neq Q(X \in A|Y=y,Z=z)\rC\]
     is a measurable $K(Y|Z)$-null set. \\
     If, furthermore, $\Xcal$ is standard then also $N := \bigcup_{A \in \Bcal_\Xcal} N_A$ is a measurable $K(Y|Z)$-null set.
     \begin{proof}
         Define $N_A^>$ and $N_A^<$ similarly with $\neq$ replaced by $>$, $<$, resp.
         Then evaluating both sides on $A \times N_A$ gives the same value. So their difference equals $0$:
         \begin{align*}
             0& =  \int \I_{N_A^>}(y,z)\cdot \lp P(X\in A|Y=y,Z=z) - Q(X \in A|Y=y,Z=z) \rp \, K(Y \in dy|Z=z), 
         \end{align*}
         where the integrand is:
         \[\I_{N_A^>}(y,z)\cdot \lp P(X\in A|Y=y,Z=z) - Q(X \in A|Y=y,Z=z) \rp  \ge 0.\]
         This implies that: $N_A^>$ must be a $K(Y|Z)$-null set.
         By symmetry we get that also $N_A^<$ is a $K(Y|Z)$-null set and thus $N_A = N_A^< \cup N_A^>$ as well.\\
         If now $\Xcal$ is a standard measurable space then $\Bcal_\Xcal =\sigma\lp \Acal\rp$ 
         with a countable set $\Acal$ that is closed under finite intersections, e.g.\ 
         $\Bcal_{[0,1]}=\sigma\lp\lC[0,c]\,|\, c \in [0,1] \cap \Q\rC\rp$. 
         One then puts $M:=\bigcup_{A \in \Acal} N_A$, which is, as countable union of $K(Y|Z)$-null sets, a $K(Y|Z)$-null set.
         Then one can define:
         \[ \Dcal := \{ A \in \Bcal_\Xcal \,|\, \forall (y,z) \in M^\cmpl:\, P(X \in A|Y=y,Z=z)=Q(X \in A|Y=y,Z=z) \}.\]
         One easily sees that $\Dcal$ is closed under complements, countable disjoint unions and contains $\Xcal \in \Dcal$.
         This shows that $\Dcal$ is a Dynkin system (aka $\lambda$-system).
         Furthermore, we have: $\Acal \ins \Dcal$ and that $\Acal$ is closed under finite intersections. 
         By Dynkin's lemma we get that:
         \[\Bcal_\Xcal = \sigma(\Acal) \ins \Dcal.\]
         This shows that $N = \bigcup_{A \in \Bcal_\Xcal}N_A \ins M$, thus equality and thus a measurable $K(Y|Z)$-null set.
     \end{proof}
 \end{Lem}

\begin{Thm}[Existence of regular conditional Markov kernels]
\label{reg-cond-trans-kernel-existence-V}
Let $(\Xcal, \Bcal_\Xcal)$ be a  standard measurable space (an analytic or a universal measurable space, resp.), 
$(\Ycal,\Bcal_\Ycal)$ a measurable space with countably generated $\Bcal_\Ycal$ 
%(e.g.\ also a standard measurable space, an analytic measurable space or a universally measurable space) 
and $(\Zcal,\Bcal_\Zcal)$ be any measurable space. 
%We might only need that $(\Ycal,\Bcal_\Ycal)$ is countable generated and/or countably separated to get map to [0,1].
Let 
$$K(X,Y|Z): \, (\Zcal, \Bcal_\Zcal) \dashrightarrow  (\Xcal \times \Ycal,\Bcal_\Xcal \otimes \Bcal_\Ycal),$$
be a Markov kernel in two variables. 
%
%A \emph{regular ($\sigma\Sous$-regular, $\bullet$-regular, resp.) conditional Markov kernel} of $K(X,Y|Z)$ conditioned on $Y$ given $Z$ is a Markov kernel:
%$$\begin{array}{rclccc}
 %K(X|Y,Z) &:&  (\Ycal \times \Zcal,& \Bcal_\Ycal\otimes \Bcal_\Zcal)& \dashrightarrow  &(\Xcal,\Bcal_\Xcal),\\
%K(X|Y,Z) &:&  (\Ycal \times \Zcal,& \sigma\Sous(\Bcal_\Ycal\otimes \Bcal_\Zcal))& \dashrightarrow  &(\Xcal,\Bcal_\Xcal),\\
%K(X|Y,Z) &:&  (\Ycal \times \Zcal,& (\Bcal_\Ycal\otimes \Bcal_\Zcal)_\bullet)& \dashrightarrow  &(\Xcal,\Bcal_\Xcal),\\
%\end{array}$$
Then there exists a regular ($\sigma\Sous$-regular, $\bullet$-regular, resp.) conditional Markov kernel $K(X|Y,Z)$,
i.e.\ measurable w.r.t.\ $\Bcal_\Ycal\otimes \Bcal_\Zcal$ (w.r.t.\ $\sigma\Sous(\Bcal_\Ycal\otimes \Bcal_\Zcal)$, w.r.t.\ $(\Bcal_\Ycal\otimes \Bcal_\Zcal)_\bullet$, resp.) 
such that for all $A \in \Bcal_\Xcal$, $B \in \Bcal_\Ycal$ and $z \in \Zcal$ we have:
$$ K(X \in A|Y=y,Z=z) 
 = \int \I_{B}(y) \cdot K(X \in A | Y=y,Z=z) \, K(dy | Z=z).$$ 
Two such conditional Markov kernels are equal up to a $K(Y|Z)$-zero set $N \in \Bcal_{\Ycal}\otimes \Bcal_\Zcal$
($N \in \sigma\Sous(\Bcal_\Ycal\otimes \Bcal_\Zcal)$, $N\in (\Bcal_\Ycal\otimes \Bcal_\Zcal)_\bullet$, resp.), see Thm. \ref{reg-cond-trans-kernel-unique}.
\begin{proof}
By assumption we find an injective Borel measurable map
$\varphi:\, (\Xcal, \Bcal_\Xcal) \inj ([0,1],\Bcal_{[0,1]})=:(\Xcal', \Bcal_{\Xcal'})$  
 that induces  a measurable isomorphism $(\Xcal, \Bcal_\Xcal) \cong (\varphi(\Xcal), \Bcal_{\Xcal|\varphi(\Xcal)})$
and $\varphi(\Xcal) \in \Bcal_{\Xcal'}$ ($\ins \Sous(\Bcal_{\Xcal'}) \ins (\Bcal_{\Xcal'})_\bullet$, resp.; for the latter we replace $\Bcal_\Xcal$ with $\Bcal_{|\Xcal}$).
So we can consider the push-forward Markov kernel $K(X',Y|Z):= K(\varphi(X),Y|Z)$:
$$K(X',Y|Z): \, (\Zcal, \Bcal_\Zcal) \stackrel{K(X,Y|Z)}{\dashrightarrow}  (\Xcal \times \Ycal,\Bcal_{\Xcal} \otimes \Bcal_{\Ycal}) 
\stackrel{\varphi \times \id_\Ycal }{\longrightarrow}  (\Xcal' \times \Ycal,\Bcal_{\Xcal'} \otimes \Bcal_{\Ycal}).$$
Since $\Xcal'=[0,1]$ and $\Bcal_\Ycal$ is countably generated we can apply Prp. \ref{reg-cond-trans-kernel-existence-II} and we then get the regular conditional Markov kernel  $K(X'|Y,Z)$:
$$K(X'|Y,Z): \, (\Ycal \times \Zcal, \Bcal_{\Ycal}\otimes \Bcal_\Zcal) \dashrightarrow  (\Xcal',\Bcal_{\Xcal'}).$$
If we put $A':=\Xcal' \sm \varphi(\Xcal)$ we have $A' \in \Bcal_{\Xcal'}$ ($\ins \sigma\Sous(\Bcal_{\Xcal'}) \ins (\Bcal_{\Xcal'})_\bullet$, resp.). 
So $K(X' \in A' |Y=y,Z=z)$ is well-defined for every $(y,z) \in \Ycal \times \Zcal$.
Consider the set:
$$D:=\{(y,z) \in \Ycal \times \Zcal | K(X' \in A'|Y=y,Z=z) >0 \}.$$
We first show that $D \in \Bcal_\Ycal \otimes \Bcal_\Zcal$ ($\ins \sigma\Sous(\Bcal_\Ycal \otimes \Bcal_\Zcal) \ins (\Bcal_\Ycal \otimes \Bcal_\Zcal)_\bullet$, resp.)
We now consider the Markov kernel $K(X'|Y,Z)$ as the measurable map:
$$ K(X'|Y,Z):\, (\Ycal \times \Zcal, \Bcal_{\Ycal}\otimes \Bcal_\Zcal) \longrightarrow (\Pcal(\Bcal_{\Xcal'}), \Bcal_{\Pcal(\Bcal_{\Xcal'})}).$$
Now consider the map:
$$ j_{A'}:\, \Pcal(\Bcal_{\Xcal'}) \to [0,1],\quad \mu \mapsto \mu(A').$$
By \ref{j-map-measurable} (or \cite{Res1977} Thm. 3+4)
the map $j_{A'}$ is measurable w.r.t.\ $\Bcal_{\Pcal(\Bcal_{\Xcal'})}$ (w.r.t.\ $\sigma\Sous(\Bcal_{\Pcal(\Bcal_{\Xcal'})})$, w.r.t.\ $(\Bcal_{\Pcal(\Bcal_{\Xcal'})})_\bullet$, resp.).
Then the composition:
$$H:\, (\Ycal \times \Zcal, \Bcal_{\Ycal}\otimes \Bcal_\Zcal) \stackrel{K(X'|Y,Z)}{\longrightarrow} 
(\Pcal(\Bcal_{\Xcal'}), \Bcal_{\Pcal(\Bcal_{\Xcal'})}) \stackrel{j_{A'}}{\longrightarrow} ([0,1],\Bcal_{[0,1]}) $$
is measurable w.r.t.\ $\Bcal_{\Ycal}\otimes \Bcal_\Zcal$ (w.r.t.\ $\sigma\Sous(\Bcal_{\Ycal}\otimes \Bcal_\Zcal)$, w.r.t.\ $(\Bcal_{\Ycal}\otimes \Bcal_\Zcal)_\bullet$, resp.).
It follows that $D = H^{-1}((0,1]) \in \Bcal_{\Ycal}\otimes \Bcal_\Zcal$ ( $\ins \sigma\Sous(\Bcal_{\Ycal}\otimes \Bcal_\Zcal) \ins (\Bcal_{\Ycal}\otimes \Bcal_\Zcal)_\bullet$, resp.)
Since $y \mapsto (y,z)$ is $\Bcal_\Ycal$-$(\Bcal_\Ycal\otimes\Bcal_\Zcal)$-measurable it follows that $D_z \in (\Bcal_\Ycal)_\bullet$.
So we can evaluate $K(Y \in D_z|Z=z)=K((Y,Z) \in D|Z=z)$ for every $z \in \Zcal$ and the map:
$$ \Zcal \stackrel{K(Y,Z|Z)}{\longrightarrow} \Pcal(\Bcal_\Ycal\otimes\Bcal_\Zcal) \stackrel{j_D}{\longrightarrow} [0,1],\quad z \mapsto K(Y \in D_z|Z=z),$$
is measurable w.r.t.\ $\Bcal_\Zcal$ (w.r.t.\ $\sigma\Sous(\Bcal_\Zcal)$, w.r.t.\ $(\Bcal_\Zcal)_\bullet$, resp.) again  by \ref{j-map-measurable} 
(or \cite{Res1977} Thm. 3+4). 
So we can integrate:
$$\begin{array}{rcl}
0 &=& K(\varphi(X) \in A', Y \in D_z |Z=z) \\
 &=& \int \I_{D_z}(y) \cdot K(X' \in A' | Y=y,Z=z) \, K(dy | Z=z). 
\end{array}$$
It follows that for all $z \in \Zcal$ we have: $K(Y \in D_z|Z=z)=0$. \\
For $A \in \Bcal_\Xcal$ let $\tilde A \in \Bcal_{\Xcal'}$ such that $A =\varphi^{-1}(\tilde A)$.
Since $\varphi$ is injective we have $\varphi(A)=\tilde A \cap f(\Xcal) \in \Bcal_{\Xcal'}$ ($\ins \sigma\Sous(\Bcal_{\Xcal'}) \ins (\Bcal_{\Xcal'})_\bullet$, resp.). 
So we can define:
$$\begin{array}{rcl} 
&& K(X \in A|Y=y,Z=z) \\
&:=& K(X' \in \varphi(A) |Y=y,Z=z) \cdot \I_{D^c}(y,z) + K_0(X \in A) \cdot \I_{D}(y,z),
\end{array}$$
with any probability distribution $K_0$.
So we get the Markov kernel:
$$K(X|Y,Z): \, (\Ycal \times \Zcal, \Bcal_{\Ycal}\otimes \Bcal_\Zcal) \dashrightarrow  (\Xcal,\Bcal_{\Xcal}),$$
(with $\sigma\Sous(\Bcal_{\Ycal}\otimes \Bcal_\Zcal)$-measurability, $(\Bcal_{\Ycal}\otimes \Bcal_\Zcal)_\bullet$-measurability, resp., again  by \ref{j-map-measurable} 
(or \cite{Res1977} Thm. 3+4)).
Furthermore, we have that for all $A \in \Bcal_\Xcal$, $B \in \Bcal_\Ycal$ and $z \in \Zcal$:
$$ K(X \in A|Y=y,Z=z) 
 = \int \I_{B}(y) \cdot K(X \in A | Y=y,Z=z) \, K(dy | Z=z).$$ 
This shows the claim. 
The uniqueness follows from Thm. \ref{reg-cond-trans-kernel-unique}.
\end{proof}
\end{Thm}
